  37 def CheckFileExt(FileName, FileExts):
  38     """Check file type based on the specified file extensions delimited by spaces.
  39     
  40     Arguments:
  41         FileName (str): Name of a file.
  42         FileExts (str): Space delimited string containing valid file extensions.
  43 
  44     Returns:
  45         bool : True, FileName contains a valid file extension; Otherwise, False.
  46 
  47     """
  48     
  49     for FileExt in FileExts.split():
  50         if re.search(r"\.%s$" % FileExt, FileName, re.IGNORECASE):
  51             return True
  52     
  53     return False

  55 def CheckTextValue(Value, ValidValues):
  56     """Check text value based on the specified valid values delimited by spaces.
  57 
  58     Arguments:
  59         Value (str): Text value
  60         ValidValues (str): Space delimited string containing valid values.
  61 
  62     Returns:
  63         bool : True, Value is valid; Otherwise, False.
  64 
  65     """
  66     
  67     ValidValues = re.sub(' ', '|', ValidValues)
  68     if re.match("^(%s)$" % ValidValues, Value, re.IGNORECASE):
  69         return True
  70     
  71     return False

  73 def GetTextLinesWords(TextFilePath, Delimiter, QuoteChar, IgnoreHeaderLine):
  74     """Parse lines in the specified text file into words in a line and return a list containing
  75     list of parsed line words.
  76 
  77     Arguments:
  78         TextFilePath (str): Text file name including file path.
  79         Delimiter (str): Delimiter for parsing text lines.
  80         QuoteChar (str): Quote character for line words.
  81         IgnoreHeaderLine (bool): A flag indicating whether to ignore first
  82             valid data line corresponding to header line.
  83 
  84     Returns:
  85         list : A list of lists containing parsed words for lines.
  86 
  87     Notes:
  88         The lines starting with # or // are considered comment lines and are
  89         ignored during parsing along with any empty lines.
  90 
  91     """
  92     TextFile = open(TextFilePath, "r")
  93     if TextFile is None:
  94         PrintError("Couldn't open text file: %s.\n" % (TextFilePath))
  95 
  96     # Collect text lines...
  97     TextLines = []
  98     FirstValidLine = True
  99     for Line in TextFile:
 100         Line = Line.rstrip()
 101         
 102         # Ignore empty lines...
 103         if not len(Line):
 104             continue
 105         
 106         # Ignore comments...
 107         if re.match("^(#|\/\/)", Line, re.I):
 108             continue
 109 
 110         # Ignore header line...
 111         if FirstValidLine:
 112             FirstValidLine = False
 113             if IgnoreHeaderLine:
 114                 continue
 115         
 116         TextLines.append(Line)
 117         
 118     TextFile.close()
 119 
 120     # Parse text lines...
 121     TextLinesWords = []
 122     
 123     TextLinesReader = csv.reader(TextLines, delimiter = Delimiter, quotechar = QuoteChar)
 124     for LineWords in TextLinesReader:
 125         TextLinesWords.append(LineWords)
 126     
 127     return TextLinesWords

 175 def ExpandFileNames(FilesSpec, Delimiter = ","):
 176     """Expand files specification using glob module to process any * or ? wild
 177     cards in file names and return a list of expanded file names.
 178 
 179     Arguments:
 180         FilesSpec (str): Files specifications
 181         Delimiter (str): Delimiter for file specifications
 182 
 183     Returns:
 184         list : List of expanded file names
 185 
 186     """
 187     FileNames = []
 188     if not len(FilesSpec):
 189         return FileNames
 190 
 191     for FileSpec in FilesSpec.split(Delimiter):
 192         FileSpec = FileSpec.strip()
 193         if re.search("(\*|\?)", FileSpec, re.I):
 194             FileNames.extend(glob.glob(FileSpec))
 195         else:
 196             FileNames.append(FileSpec)
 197     
 198     return FileNames

 200 def GetExamplesTextFromDocOptText(DocOptText):
 201     """Get script usage example lines from a docopt doc string. The example text
 202     line start from a line containing `Examples:`  keyword at the beginning of the line.
 203     
 204     Arguments:
 205         DocOptText (str): Doc string containing script usage examples lines starting with
 206             a line marked by `Examples:` keyword at the beginning of a line.
 207 
 208     Returns:
 209         str : A string containing text lines retrieved from the examples section of
 210             DocOptText parameter.
 211 
 212     """
 213     
 214     ExamplesStart = re.compile("^Examples:", re.IGNORECASE)
 215     ExamplesEnd = re.compile("^(Author:|See also:|Copyright:)", re.IGNORECASE)
 216     
 217     ExamplesText = 'Examples text is not available';
 218     ExamplesTextFound = False
 219     
 220     for Line in DocOptText.splitlines():
 221         if ExamplesStart.match(Line):
 222             ExamplesText = 'Examples:';
 223             ExamplesTextFound = True
 224             continue
 225         
 226         if ExamplesEnd.match(Line):
 227             break
 228         
 229         if ExamplesTextFound:
 230             ExamplesText += "\n" + Line;
 231     
 232     return ExamplesText

 234 def GetExcelStyleColumnLabel(ColNum):
 235     """Return Excel style column label for a colum number.
 236     
 237     Arguments:
 238         ColNum (int): Column number
 239 
 240     Returns:
 241         str : Excel style column label.
 242 
 243     """
 244     Letters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'
 245     
 246     ColLabelList = []

 308 def GetFormattedFileSize(FileName, Precision = 1):
 309     """Get file size  as a string in the following format: %.*f <bytes, KB, MB,
 310     GB>
 311     
 312     Arguments:
 313         FileName (str): File path.
 314         Precision (int): File size precision.
 315 
 316     Returns:
 317         str : File size formatted as: %.2f <bytes, KB, MB, GB>
 318 
 319     """
 320     
 321     Size = os.path.getsize(FileName)
 322 
 323     if Size < 1024:
 324         SizeDenominator = 1
 325         SizeSuffix = "bytes"
 326     elif Size < (1024*1024):
 327         SizeDenominator = 1024
 328         SizeSuffix = "KB"
 329     elif Size < (1024*1024*1024):
 330         SizeDenominator = 1024*1024
 331         SizeSuffix = "MB"
 332     elif Size < (1024*1024*1024*1024):
 333         SizeDenominator = 1024*1024*1024
 334         SizeSuffix = "GB"
 335     else:
 336         SizeDenominator = 1
 337         SizeSuffix = "bytes"
 338         
 339     Size /= SizeDenominator
 340 
 341     FormattedSize = "%.*f %s" % (Precision, Size, SizeSuffix)
 342     
 343     return FormattedSize

 590 def ValidateOptionNumberValue(OptionName, OptionValue, CmpOpValueMap):
 591     """Validate option value using comparison operater and value pairs in specified in
 592     a map.
 593     
 594     Arguments:
 595         OptionName (str): Command line option name.
 596         OptionValue (int or float): Command line option value.
 597         CmpOpValueMap (dictionary): Comparison operator key and value pairs to
 598             validate values specified in OptionValue.
 599 
 600     Notes:
 601         The function exits with an error message for an invalid option values specified
 602         in OptionValue.
 603 
 604     Examples:
 605 
 606         ValidateOptionNumberValue("--maxConfs", int(Options["--maxConfs"]),
 607             {">": 0})
 608         ValidateOptionNumberValue("-b, --butinaSimilarityCutoff", 
 609             float(Options["--butinaSimilarityCutoff"]),
 610             {">": 0.0, "<=" : 1.0})
 611 
 612     """
 613     
 614     Status = True
 615     for CmpOp in CmpOpValueMap:
 616         Value = CmpOpValueMap[CmpOp]
 617         if re.match("^>$", CmpOp, re.I):
 618             if OptionValue <= Value:
 619                 Status = False
 620                 break
 621         elif re.match("^>=$", CmpOp, re.I):
 622             if OptionValue < Value:
 623                 Status = False
 624                 break
 625         elif re.match("^<$", CmpOp, re.I):
 626             if OptionValue >= Value:
 627                 Status = False
 628                 break
 629         elif re.match("^<=$", CmpOp, re.I):
 630             if OptionValue > Value:
 631                 Status = False
 632                 break
 633         else:
 634             PrintError("The specified comparison operator, %s, for function MiscUtil.ValidateOptionNumberValue is not supported\n" % (CmpOp))
 635     
 636     if not Status:
 637         FirstValue = True
 638         SupportedValues = ""
 639         for CmpOp in CmpOpValueMap:
 640             Value = CmpOpValueMap[CmpOp]
 641             if FirstValue:
 642                 FirstValue = False
 643                 SupportedValues = "%s %s" % (CmpOp, Value)
 644             else:
 645                 SupportedValues = "%s and %s %s" % (SupportedValues, CmpOp, Value)
 646         
 647         PrintError("The value specified, %s, for option \"%s\" is not valid. Supported value(s): %s " % (OptionValue, OptionName, SupportedValues))
 648 
 649 def ValidateOptionNumberValues(OptionName, OptionValueString, OptionValueCount, OptionValueDelimiter, OptionValueType, CmpOpValueMap):
 650     """Validate numerical option values using option value string, delimiter, value type,
 651     and a specified map containing comparison operator and value pairs.
 652     
 653     Arguments:
 654         OptionName (str): Command line option name.
 655         OptionValueString (str): Command line option value.
 656         OptionValueCount (int): Number of values in OptionValueString.
 657         OptionValueDelimiter (str): Delimiter used for values in OptionValueString.
 658         OptionValueType (str): Valid number types (integer or float)
 659         CmpOpValueMap (dictionary): Comparison operator key and value pairs to
 660             validate values specified in OptionValueString.
 661 
 662     Notes:
 663         The function exits with an error message for invalid option values specified
 664         in OptionValueString
 665 
 666     Examples:
 667 
 668         ValidateOptionNumberValues("-m, --molImageSize",
 669             Options["--molImageSize"], 2, ",", "integer", {">": 0})
 670 
 671     """
 672     if not CheckTextValue(OptionValueType, "integer float"):
 673         PrintError("The option value type specified, %s, for function MiscUtil.ValidateOptionNumberValues  is not valid. Supported value: integer float " % (OptionValueType))
 674         
 675     Values = OptionValueString.split(OptionValueDelimiter)
 676     if len(Values) != OptionValueCount:
 677         PrintError("The value specified, %s, for option \"%s\" is not valid. It must contain %d %s values separated by \"%s\"" % (OptionValueString, OptionName, OptionValueCount, OptionValueType, OptionValueDelimiter))
 678 
 679     IsIntergerValue = True
 680     if re.match("^float$", OptionValueType, re.I):
 681         IsIntergerValue = False
 682     
 683     for Value in Values:
 684         if IsIntergerValue:
 685             if not IsInteger(Value):
 686                 PrintError("The value specified, %s, for option \"%s\" in string \"%s\" must be an integer." % (Value, OptionName, OptionValueString))
 687             Value = int(Value)
 688         else:
 689             if not IsFloat(Value):
 690                 PrintError("The value specified, %s, for option \"%s\" in string \"%s\" must be a float." % (Value, OptionName, OptionValueString))
 691             Value = float(Value)
 692         ValidateOptionNumberValue(OptionName, Value, CmpOpValueMap)
 693

 802 def _ProcessInfileAndOutfileParameters(Mode, ParamsInfo, ParamsOptionName, ParamsOptionValue, InfileName, OutfileName):
 803     """Process specified infile and outfile paramaters.
 804     
 805     """
 806     if re.match("^auto$", ParamsOptionValue, re.I):
 807         # No specific parameters to process except for parameters with possible auto value...
 808         _ProcessInfileAndOutfileAutoParameters(Mode, ParamsInfo, ParamsOptionName, ParamsOptionValue, InfileName, OutfileName)
 809         return
 810     
 811     ParamsOptionValue = re.sub(" ", "", ParamsOptionValue)
 812     if not ParamsOptionValue:
 813         PrintError("No valid parameter name and value pairs specified using \"%s\" option" % ParamsOptionName)
 814     
 815     ParamsOptionValueWords = ParamsOptionValue.split(",")
 816     if len(ParamsOptionValueWords) % 2:
 817         PrintError("The number of comma delimited paramater names and values, %d, specified using \"%s\" option must be an even number." % (len(ParamsOptionValueWords), ParamsOptionName))
 818         
 819     # Setup a canonical paramater names...
 820     ValidParamNames = []
 821     CanonicalParamNamesMap = {}
 822     for ParamName in sorted(ParamsInfo):
 823         ValidParamNames.append(ParamName)
 824         CanonicalParamNamesMap[ParamName.lower()] = ParamName
 825     
 826     # Validate paramater name and value pairs...
 827     for Index in range(0, len(ParamsOptionValueWords), 2):
 828         Name = ParamsOptionValueWords[Index]
 829         Value = ParamsOptionValueWords[Index + 1]
 830 
 831         CanonicalName = Name.lower()
 832         if  not CanonicalName in CanonicalParamNamesMap:
 833             PrintError("The parameter name, %s, specified using \"%s\" is not a valid name. Supported parameter names: %s" % (Name, ParamsOptionName, " ".join(ValidParamNames)))
 834 
 835         ParamName = CanonicalParamNamesMap[CanonicalName]
 836         ParamValue = Value
 837         
 838         if re.match("^(Sanitize|StrictParsing|RemoveHydrogens|Kekulize|SMILESIsomeric)$", ParamName, re.I):
 839             if not re.match("^(Yes|No|True|False)$", Value, re.I):
 840                 PrintError("The parameter value, %s, specified for parameter name, %s, using \"%s\" option is not a valid value. Supported values: Yes No True False" % (Value, Name, ParamsOptionName))
 841             ParamValue = True
 842             if re.match("^(No|False)$", Value, re.I):
 843                 ParamValue = False
 844         elif re.match("^SMILESTitleLine$", ParamName, re.I):
 845             if re.match("^Infile$", Mode, re.I):
 846                 if not re.match("^(Yes|No|True|False|Auto)$", Value, re.I):
 847                     PrintError("The parameter value, %s, specified for paramater name, %s, using \"%s\" option is not a valid value. Supported values: Yes No True False Auto" % (Value, Name, ParamsOptionName))
 848             elif re.match("^Outfile$", Mode, re.I):
 849                 if not re.match("^(Yes|No|True|False)$", Value, re.I):
 850                     PrintError("The parameter value, %s, specified for parameter name, %s, using \"%s\" option is not a valid value. Supported values: Yes No True False" % (Value, Name, ParamsOptionName))
 851                 ParamValue = True
 852                 if re.match("^(No|False)$", Value, re.I):
 853                     ParamValue = False
 854         elif re.match("^SMILESDelimiter$", ParamName, re.I):
 855             if not re.match("^(space|tab|comma)$", Value, re.I):
 856                 PrintError("The parameter value, %s, specified for parameter name, %s, using \"%s\" option is not a valid value. Supported values: space tab comma" % (Value, Name, ParamsOptionName))
 857             ParamValue = " "
 858             if re.match("^tab$", Value, re.I):
 859                 ParamValue = "\t"
 860             elif re.match("^comma$", Value, re.I):
 861                 ParamValue = ","
 862         elif re.match("^Compute2DCoords$", ParamName, re.I):
 863             # No need to set the value. It would be processed later to handle "auto" value...
 864             if not re.match("^(Yes|No|True|False|Auto)$", Value, re.I):
 865                 PrintError("The parameter value, %s, specified for paramater name, %s, using \"%s\" option is not a valid value. Supported values: Yes No True False Auto" % (Value, Name, ParamsOptionName))
 866         else:
 867             ParamValue = int(Value)
 868             if ParamValue <= 0:
 869                 PrintError("The parameter value, %s, specified for parameter name, %s, using \"%s\" option is not a valid value. Supported values: > 0" % (Value, Name, ParamsOptionName))
 870         
 871         # Set value...
 872         ParamsInfo[ParamName] = ParamValue
 873         
 874     # Handle paramaters with possible auto values...
 875     _ProcessInfileAndOutfileAutoParameters(Mode, ParamsInfo, ParamsOptionName, ParamsOptionValue, InfileName, OutfileName)

 877 def _ProcessInfileAndOutfileAutoParameters(Mode, ParamsInfo, ParamsOptionName, ParamsOptionValue, InfileName, OutfileName):
 878     """Process parameters with possible auto values.
 879     
 880     """
 881     if re.match("^Infile$", Mode, re.I):
 882         # SMILESTitleLine parameter...
 883         Value = ParamsInfo["SMILESTitleLine"]
 884         ParamValue = False
 885         if re.match("^auto$", Value, re.I):
 886             if InfileName is not None:
 887                 if CheckFileExt(InfileName, "smi csv tsv txt"):
 888                     ParamValue = DoesSMILESFileContainTitleLine(InfileName)
 889         elif re.match("^(Yes|True)$", Value, re.I):
 890             ParamValue = True
 891         ParamsInfo["SMILESTitleLine"] = ParamValue
 892     elif re.match("^Outfile$", Mode, re.I):
 893         # Compute2DCoords parameter...
 894         Value = ParamsInfo["Compute2DCoords"]
 895         ParamValue = False
 896         if re.match("^auto$", Value, re.I):
 897             if InfileName is not None:
 898                 if CheckFileExt(InfileName, "smi csv tsv txt"):
 899                     ParamValue = True
 900             if OutfileName is not None:
 901                 if CheckFileExt(OutfileName, "smi csv tsv txt"):
 902                     # No need to compute 2D coords for SMILES file...
 903                     ParamValue = False
 904         elif re.match("^(Yes|True)$", Value, re.I):
 905             ParamValue = True
 906         ParamsInfo["Compute2DCoords"] = ParamValue
