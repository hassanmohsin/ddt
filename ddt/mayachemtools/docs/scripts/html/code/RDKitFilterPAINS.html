  81 def PerformFiltering():
  82     """Filter molecules using SMARTS specified in PAINS filter file."""
  83     
  84     Infile = OptionsInfo["Infile"]
  85     Outfile = OptionsInfo["Outfile"]
  86     
  87     CountMode = OptionsInfo["CountMode"]
  88     NegateMatch = OptionsInfo["NegateMatch"]
  89     PAINSMode = OptionsInfo["PAINSMode"]
  90 
  91     # Setup PAINS patterns and pattern mols...
  92     PAINSPatterns = RetrievePAINSPatterns(PAINSMode)
  93     PAINSPatternMols = SetupPAINSPatternMols(PAINSPatterns)
  94     
  95     # Setup a molecule reader...
  96     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
  97     Mols  = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
  98     
  99     # Set up a molecule writer...
 100     Writer = None
 101     if not CountMode:
 102         Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
 103         if Writer is None:
 104             MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
 105         MiscUtil.PrintInfo("Generating file %s..." % Outfile)
 106 
 107     Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
 108     
 109     # Process molecules...
 110     MolCount = 0
 111     ValidMolCount = 0
 112     FilteredCount = 0
 113 
 114     for Mol in Mols:
 115         MolCount += 1
 116         
 117         if Mol is None:
 118             continue
 119         
 120         if RDKitUtil.IsMolEmpty(Mol):
 121             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 122             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 123             continue
 124         
 125         ValidMolCount += 1
 126         
 127         MolMatched = DoesMoleculeContainsPAINSPattern(Mol, PAINSPatternMols)
 128         if MolMatched == NegateMatch:
 129             FilteredCount += 1
 130             if not CountMode:
 131                 if Compute2DCoords:
 132                     AllChem.Compute2DCoords(Mol)
 133                 Writer.write(Mol)
 134     
 135     if Writer is not None:
 136         Writer.close()
 137     
 138     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 139     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 140     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount))
 141 
 142     MiscUtil.PrintInfo("\nTotal number of filtered molecules: %d" % FilteredCount)
 143 
 144 def DoesMoleculeContainsPAINSPattern(Mol, PAINSPatternMols):
 145     """Check presence of PAINS pattern in the molecule"""
 146 
 147     MolMatched = False
 148     
 149     for PatternMol in PAINSPatternMols:
 150         if Mol.HasSubstructMatch(PatternMol, useChirality = True):
 151             MolMatched = True
 152             break
 153         
 154     return MolMatched
 155     
 156 def RetrievePAINSPatterns(PAINSFilterMode):
 157     """Retrieve PAINS patterns for specified PAINS mode"""
 158 
 159     MayaChemToolsDataDir = MiscUtil.GetMayaChemToolsLibDataPath()
 160     PAINSFiltersFilePath = os.path.join(MayaChemToolsDataDir, "PAINSFilters.csv")
 161     
 162     MiscUtil.PrintInfo("\nRetrieving PAINS SMARTS patterns for PAINS filter type, %s, from file %s" % (PAINSFilterMode, PAINSFiltersFilePath))
 163 
 164     if not os.path.exists(PAINSFiltersFilePath):
 165         MiscUtil.PrintError("The PAINS filters file, %s, doesn't exist.\n" % (PAINSFiltersFilePath))
 166         
 167     FilterFile = open(PAINSFiltersFilePath, "r")
 168     if FilterFile is None:
 169         MiscUtil.PrintError("Couldn't open PAINS filter file: %s.\n" % (PAINSFiltersFilePath))
 170 
 171     # Collect all PAINS filter lines...
 172     HeaderLine = True
 173     FiltersLines = []
 174     for Line in FilterFile:
 175         Line = Line.rstrip()
 176         # Ignore comments...
 177         if re.match("^#", Line, re.I):
 178             continue
 179         # Ignore header line...
 180         if HeaderLine:
 181             HeaderLine = False
 182             continue
 183         FiltersLines.append(Line)
 184         
 185     # Process PAINS filter lines using csv reader...
 186     SMARTSPatterns = []
 187     
 188     FiltersReader = csv.reader(FiltersLines, delimiter=',', quotechar='"')
 189     for LineWords in FiltersReader:
 190         FilterType = LineWords[0]
 191         ID = LineWords[1]
 192         SMARTS = LineWords[2]
 193 
 194         if re.match("^All$", PAINSFilterMode, re.I) or FilterType.lower() == PAINSFilterMode.lower():
 195             SMARTSPatterns.append(SMARTS)
 196         
 197     FilterFile.close()
 198 
 199     MiscUtil.PrintInfo("Total number of PAINS SMARTS patterns: %d" % (len(SMARTSPatterns)))
 200     
 201     return SMARTSPatterns
 202 
 203 def SetupPAINSPatternMols(PAINSPatterns):
 204     """Set up PAINS pattern mols for substructure search"""
 205 
 206     MiscUtil.PrintInfo("\nSetting up PAINS pattern molecules for performins substructure search...")
 207     PatternMols = []
 208     for Pattern in PAINSPatterns:
 209         PatternMol = Chem.MolFromSmarts(Pattern)
 210         PatternMols.append(PatternMol)
 211         
 212     return PatternMols

 214 def ProcessOptions():
 215     """Process and validate command line arguments and options"""
 216     
 217     MiscUtil.PrintInfo("Processing options...")
 218     
 219     # Validate options...
 220     ValidateOptions()
 221     
 222     OptionsInfo["Infile"] = Options["--infile"]
 223     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 224     
 225     OptionsInfo["Outfile"] = Options["--outfile"]
 226     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"], Options["--infile"], Options["--outfile"])
 227     
 228     OptionsInfo["Overwrite"] = Options["--overwrite"]
 229 
 230     OptionsInfo["CountMode"] = False
 231     if re.match("^count$", Options["--mode"], re.I):
 232         OptionsInfo["CountMode"] = True
 233     
 234     OptionsInfo["NegateMatch"] = False
 235     if re.match("^yes$", Options["--negate"], re.I):
 236         OptionsInfo["NegateMatch"] = True
 237         
 238     OptionsInfo["PAINSMode"] = Options["--painsMode"]
 239     
 240 def RetrieveOptions():
 241     """Retrieve command line arguments and options"""
 242     
 243     # Get options...
 244     global Options
 245     Options = docopt(_docoptUsage_)
 246     
 247     # Set current working directory to the specified directory...
 248     WorkingDir = Options["--workingdir"]
 249     if WorkingDir:
 250         os.chdir(WorkingDir)
 251     
 252     # Handle examples option...
 253     if "--examples" in Options and Options["--examples"]:
 254         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(_docoptUsage_))
 255         sys.exit(0)

 257 def ValidateOptions():
 258     """Validate option values"""
 259     
 260     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 261     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd smi txt csv tsv")
 262     
 263     MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd smi")
 264     MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 265     MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 266 
 267     MiscUtil.ValidateOptionTextValue("-m, --mode", Options["--mode"], "filter count")
 268     if re.match("^filter$", Options["--mode"], re.I):
 269         if not Options["--outfile"]:
 270             MiscUtil.PrintError("The outfile must be specified using \"-o, --outfile\" during \"filter\" value of \"-m, --mode\" option")
 271         
 272     MiscUtil.ValidateOptionTextValue("-n, --negate", Options["--negate"], "yes no")
 273     
 274     MiscUtil.ValidateOptionTextValue("-p, --painsMode", Options["--painsMode"], "All A B C")
