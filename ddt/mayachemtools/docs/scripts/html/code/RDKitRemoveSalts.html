  82 def RemoveSalts():
  83     """Identify and remove salts from molecules"""
  84     
  85     Infile = OptionsInfo["Infile"]
  86     Outfile = OptionsInfo["Outfile"]
  87     
  88     CountMode = OptionsInfo["CountMode"]
  89     
  90     # Setup a molecule reader...
  91     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
  92     Mols  = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
  93     
  94     # Set up a molecule writer...
  95     Writer = None
  96     DuplicatesWriter = None
  97     if not CountMode:
  98         Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
  99         if Writer is None:
 100             MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
 101         
 102         MiscUtil.PrintInfo("Generating file %s..." % (Outfile))
 103 
 104     # Set up a salt remover...
 105     SaltsByComponentsMode = OptionsInfo["SaltsByComponentsMode"]
 106     Remover = None
 107     if not SaltsByComponentsMode:
 108         Remover = SaltRemover(defnFilename = OptionsInfo["SaltsFile"], defnData = OptionsInfo["SaltsSMARTS"], defnFormat = InputFormat.SMARTS)
 109     
 110     # Process molecules...
 111     MolCount = 0
 112     ValidMolCount = 0
 113     SaltsMolCount = 0
 114     
 115     Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
 116 
 117     for Mol in Mols:
 118         MolCount += 1
 119         
 120         if Mol is None:
 121             continue
 122         
 123         if RDKitUtil.IsMolEmpty(Mol):
 124             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 125             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 126             continue
 127         
 128         ValidMolCount += 1
 129         
 130         UnsaltedMol, SaltyStatus = RemoveMolSalts(Mol, Remover, MolCount)
 131         
 132         if SaltyStatus:
 133             SaltsMolCount += 1
 134 
 135         if not CountMode:
 136             if Compute2DCoords:
 137                 AllChem.Compute2DCoords(UnsaltedMol)
 138             Writer.write(UnsaltedMol)
 139     
 140     if Writer is not None:
 141         Writer.close()
 142     
 143     if DuplicatesWriter is not None:
 144         DuplicatesWriter.close()
 145     
 146     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 147     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 148     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount))
 149     
 150     MiscUtil.PrintInfo("\nNumber of molecules coontaining salts: %d" % (SaltsMolCount))

 152 def RemoveMolSalts(Mol, Remover, MolCount):
 153     """Remove salts from mol and return unsalted mol along with mol salty status."""
 154 
 155     UnsaltedMol = Mol
 156     SaltyStatus = False
 157     
 158     if Remover is not None:
 159         KeptMol, DeletedMols = Remover.StripMolWithDeleted(Mol, dontRemoveEverything = False)
 160         if len(DeletedMols) >= 1:
 161             SaltyStatus = True
 162         if RDKitUtil.IsMolEmpty(KeptMol):
 163             if len(DeletedMols) >= 1:
 164                 # Take the larged fragment from DeletedMols
 165                 UnsaltedMol = GetLargestMol(DeletedMols)
 166     else:
 167         # Use largest fragment as unsalted molecule...
 168         MolFrags = Chem.GetMolFrags(Mol, asMols = True)
 169         if len(MolFrags) > 1:
 170             # Keep the largest fragment as unsalted molecule...
 171             SaltyStatus = True
 172             UnsaltedMol = GetLargestMol(MolFrags)
 173 
 174     if SaltyStatus:
 175         Chem.SanitizeMol(UnsaltedMol)
 176         MolName = RDKitUtil.GetMolName(Mol, MolCount)
 177         if len(MolName):
 178             UnsaltedMol.SetProp("_Name", MolName)
 179     
 180     return (UnsaltedMol, SaltyStatus)
 181 
 182 def GetLargestMol(Mols):
 183     """Get largest mol from list of mols"""
 184 
 185     LargestMol = None
 186     LargestMolSize = -1
 187     for Mol in Mols:
 188         Size = Mol.GetNumAtoms()
 189         if Size > LargestMolSize:
 190             LargestMol = Mol
 191             LargestMolSize = Size
 192 
 193     return LargestMol

 195 def ProcessOptions():
 196     """Process and validate command line arguments and options"""
 197     
 198     MiscUtil.PrintInfo("Processing options...")
 199     
 200     # Validate options...
 201     ValidateOptions()
 202     
 203     OptionsInfo["Infile"] = Options["--infile"]
 204     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 205     
 206     OptionsInfo["Outfile"] = Options["--outfile"]
 207     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"], Options["--infile"], Options["--outfile"])
 208 
 209     OptionsInfo["Overwrite"] = Options["--overwrite"]
 210 
 211     OptionsInfo["CountMode"] = False
 212     if re.match("^count$", Options["--mode"], re.I):
 213         OptionsInfo["CountMode"] = True
 214 
 215     SaltsByComponentsMode = False
 216     SaltsBySMARTSFileMode = False
 217     SaltsBySMARTSMode = False
 218     if re.match("^ByComponent$", Options["--saltsMode"], re.I):
 219         SaltsByComponentsMode = True
 220     elif re.match("^BySMARTSFile$", Options["--saltsMode"], re.I):
 221         SaltsBySMARTSFileMode = False
 222     elif re.match("^BySMARTS$", Options["--saltsMode"], re.I):
 223         SaltsBySMARTSMode = True
 224     else:
 225         MiscUtil.PrintError("The salts mode specified, %s, using \"--saltsMode\" option is not valid." % Options["--saltsMode"])
 226     OptionsInfo["SaltsByComponentsMode"]  = SaltsByComponentsMode
 227     OptionsInfo["SaltsBySMARTSFileMode"]  = SaltsBySMARTSFileMode
 228     OptionsInfo["SaltsBySMARTSMode"]  = SaltsBySMARTSMode
 229 
 230     SaltsFile = None
 231     if re.match("^BySMARTSFile$", Options["--saltsMode"], re.I):
 232         if not re.match("^auto$", Options["--saltsFile"], re.I):
 233             SaltsFile = Options["--saltsFile"]
 234     OptionsInfo["SaltsFile"] = SaltsFile
 235     
 236     SaltsSMARTS = None
 237     if re.match("^BySMARTS$", Options["--saltsMode"], re.I):
 238         if not Options["--saltsSMARTS"]:
 239             MiscUtil.PrintError("No salts SMARTS pattern specified using \"--saltsSMARTS\" option during \"BySMARTS\" value of \"-s, --saltsMode\" option")
 240         SaltsSMARTS = Options["--saltsSMARTS"].strip(" ")
 241         if not len(SaltsSMARTS):
 242             MiscUtil.PrintError("Empty SMARTS pattern specified using \"--saltsSMARTS\" option during \"BySMARTS\" value of \"-s, --saltsMode\" option")
 243         if re.search(" ", SaltsSMARTS):
 244             SaltsSMARTS = re.sub('[ ]+', '\n', SaltsSMARTS)
 245         
 246     OptionsInfo["SaltsSMARTS"] = SaltsSMARTS
 247     
 248 def RetrieveOptions():
 249     """Retrieve command line arguments and options"""
 250     
 251     # Get options...
 252     global Options
 253     Options = docopt(_docoptUsage_)
 254     
 255     # Set current working directory to the specified directory...
 256     WorkingDir = Options["--workingdir"]
 257     if WorkingDir:
 258         os.chdir(WorkingDir)
 259     
 260     # Handle examples option...
 261     if "--examples" in Options and Options["--examples"]:
 262         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(_docoptUsage_))
 263         sys.exit(0)
 264 
 265 def ValidateOptions():
 266     """Validate option values"""
 267     
 268     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 269     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd smi txt csv tsv")
 270     
 271     if Options["--outfile"]:
 272         MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd smi")
 273         MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 274         MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 275 
 276     MiscUtil.ValidateOptionTextValue("-m, --mode", Options["--mode"], "remove count")
 277     if re.match("^remove$", Options["--mode"], re.I):
 278         if not Options["--outfile"]:
 279             MiscUtil.PrintError("The outfile must be specified using \"-o, --outfile\" during \"remove\" value of \"-m, --mode\" option")
 280     
 281     MiscUtil.ValidateOptionTextValue("--saltsMode", Options["--saltsMode"], "ByComponent BySMARTSFile BySMARTS")
 282     
 283     if re.match("^BySMARTSFile$", Options["--saltsMode"], re.I):
 284         if not re.match("^auto$", Options["--saltsFile"], re.I):
 285             MiscUtil.ValidateOptionFilePath("--saltsFile", Options["--saltsFile"])
