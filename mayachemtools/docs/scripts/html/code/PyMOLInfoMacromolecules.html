 118 def ListHeaderInfo(Infile):
 119     """List header information."""
 120 
 121     if not OptionsInfo["Header"]:
 122         return
 123 
 124     FileDir, FileName, FileExt = MiscUtil.ParseFileName(Infile)
 125 
 126     Classification, DepositionDate, IDCode, ExperimentalTechnique, Resolution = ["Not Available"] * 5
 127     if re.match("^pdb$", FileExt, re.I):
 128          Classification, DepositionDate, IDCode, ExperimentalTechnique, Resolution = RetriveHeadAndExperimentalInfoFromPDBFile(Infile)
 129     elif re.match("^cif$", FileExt, re.I):
 130          Classification, DepositionDate, IDCode, ExperimentalTechnique, Resolution = RetriveHeadAndExperimentalInfoFromCIFFile(Infile)
 131 
 132     MiscUtil.PrintInfo("\nID: %s\nClassification: %s\nDesposition date: %s" % (IDCode, Classification, DepositionDate))
 133     MiscUtil.PrintInfo("\nExperimental technique %s\nResolution: %s" % (ExperimentalTechnique, Resolution))

 135 def RetriveHeadAndExperimentalInfoFromPDBFile(Infile):
 136     """Retrieve header and experimental information from PDB file. """
 137 
 138     Classification, DepositionDate, IDCode, ExperimentalTechnique, Resolution = ["Not Available"] * 5
 139 
 140     Lines = MiscUtil.GetTextLines(Infile)
 141 
 142     # Retrieve header info...
 143     for Line in Lines:
 144         if re.match("^HEADER", Line, re.I):
 145             # Format: 10x40s9s3x4s
 146             FormatSize = 66
 147             Line = PrepareLineForFormatSize(Line, FormatSize)
 148             
 149             Classification = Line[10:50]
 150             DepositionDate = Line[50:59]
 151             IDCode = Line[62:66]
 152             
 153             Classification = Classification.strip() if len(Classification.strip()) else "Not Available"
 154             DepositionDate = DepositionDate.strip() if len(DepositionDate.strip()) else "Not Available"
 155             IDCode = IDCode.strip() if len(IDCode.strip()) else "Not Available"
 156             
 157             break
 158 
 159     # Retrieve experimental info...
 160     for Line in Lines:
 161         if re.match("^EXPDTA", Line, re.I):
 162             ExperimentalTechnique = re.sub("^EXPDTA", "", Line, re.I)
 163             ExperimentalTechnique = ExperimentalTechnique.strip()
 164         elif re.match("^REMARK   2 RESOLUTION.", Line, re.I):
 165             if re.search("NOT APPLICABLE", Line, re.I):
 166                 Resolution = "NOT APPLICABLE"
 167             else:
 168                 FormatSize = 70
 169                 Line = PrepareLineForFormatSize(Line, FormatSize)
 170                 Resolution = Line[22:70]
 171                 Resolution = Resolution.strip() if len(Resolution.strip()) else "Not Available"
 172         elif re.match("^(ATOM|HETATM)", Line, re.I):
 173             break
 174 
 175     return Classification, DepositionDate, IDCode, ExperimentalTechnique, Resolution
 176 
 177 def RetriveHeadAndExperimentalInfoFromCIFFile(Infile):
 178     """Retrieve header information from CIF file. """
 179 
 180     Classification, DepositionDate, IDCode, ExperimentalTechnique, Resolution = ["Not Available"] * 5
 181     
 182     Lines = MiscUtil.GetTextLines(Infile)
 183     
 184     # IDCode...
 185     for Line in Lines:
 186         if re.match("^_struct_keywords.entry_id", Line, re.I):
 187             IDCode = re.sub("^_struct_keywords.entry_id", "", Line, re.I)
 188             IDCode = IDCode.strip() if len(IDCode.strip()) else "Not Available"
 189             break
 190 
 191     # Classification...
 192     for Line in Lines:
 193         if re.match("^_struct_keywords.pdbx_keywords", Line, re.I):
 194             Classification = re.sub("^_struct_keywords.pdbx_keywords", "", Line, re.I)
 195             Classification = Classification.strip() if len(Classification.strip()) else "Not Available"
 196             break
 197     
 198     # Deposition date...
 199     for Line in Lines:
 200         if re.match("^_pdbx_database_status.recvd_initial_deposition_date", Line, re.I):
 201             DepositionDate = re.sub("^_pdbx_database_status.recvd_initial_deposition_date", "", Line, re.I)
 202             DepositionDate = DepositionDate.strip() if len(DepositionDate.strip()) else "Not Available"
 203             break
 204     
 205     # Experimental technique...
 206     for Line in Lines:
 207         if re.match("^_exptl.method", Line, re.I):
 208             ExperimentalTechnique = re.sub("(_exptl.method|')", "", Line, flags = re.I)
 209             ExperimentalTechnique = ExperimentalTechnique.strip() if len(ExperimentalTechnique.strip()) else "Not Available"
 210             break
 211         
 212     # Resolution...
 213     for Line in Lines:
 214         if re.match("^_reflns.d_resolution_high", Line, re.I):
 215             Resolution = re.sub("^_reflns.d_resolution_high", "", Line, re.I)
 216             Resolution = Resolution.strip() if len(Resolution.strip()) else "Not Available"
 217             break
 218         
 219     return Classification, DepositionDate, IDCode, ExperimentalTechnique, Resolution

 221 def PrepareLineForFormatSize(Text, FormatSize):
 222     """Prepare text string for format size padding or truncation to
 223     alter its size.
 224     """
 225 
 226     TextLen = len(Text)
 227     if TextLen < FormatSize:
 228         PaddingLen = FormatSize - TextLen
 229         TextPadding = " " * PaddingLen
 230         Text = Text + TextPadding
 231     elif TextLen > FormatSize:
 232         Text = Text[:FormatSize]
 233     
 234     return Text

 279 def ListSelectionResiduesInfo(MolName, ChainIDs, SelectionType):
 280     """List residues information for a specified selection type. """
 281 
 282     Lines = []
 283     TotalResCount = 0
 284     
 285     for ChainID in ChainIDs:
 286         SelectionInfo, SelectionLabel = GetSelectionResiduesInfo(MolName, ChainID, SelectionType)
 287 
 288         ChainResCount = 0
 289         LineWords = []
 290         
 291         SortedResNames = sorted(SelectionInfo["ResNames"], key = lambda ResName: SelectionInfo["ResCount"][ResName], reverse = True)
 292         for ResName in SortedResNames:
 293             ResCount = SelectionInfo["ResCount"][ResName]
 294             LineWords.append("%s - %s" % (ResName, ResCount))
 295 
 296             ChainResCount += ResCount
 297             TotalResCount += ResCount
 298         
 299         Line = "; ".join(LineWords) if len(LineWords) else None
 300         Lines.append("Chain ID: %s; Count: %s; Names:  %s" % (ChainID, ChainResCount, Line))
 301     
 302     MiscUtil.PrintInfo("\nNumber of %s residues: %s" % (SelectionLabel, TotalResCount))
 303     for Line in Lines:
 304         MiscUtil.PrintInfo("%s" % Line)

 306 def GetSelectionResiduesInfo(MolName, ChainID, SelectionType):
 307     """Get residues info for a specified selection type. """
 308 
 309     SelectionInfo = None
 310     SelectionLabel = None
 311     
 312     if re.match("^Ligands$", SelectionType, re.I):
 313         SelectionLabel = "ligand"
 314         SelectionInfo = PyMOLUtil.GetLigandResiduesInfo(MolName, ChainID)
 315     elif re.match("^Solvents$", SelectionType, re.I):
 316         SelectionLabel = "solvent"
 317         SelectionInfo = PyMOLUtil.GetSolventResiduesInfo(MolName, ChainID)
 318     elif re.match("^Inorganics$", SelectionType, re.I):
 319         SelectionLabel = "inorganic"
 320         SelectionInfo = PyMOLUtil.GetInorganicResiduesInfo(MolName, ChainID)
 321     elif re.match("^Chains$", SelectionType, re.I):
 322         SelectionLabel = "polymer chain"
 323         SelectionInfo = PyMOLUtil.GetPolymerResiduesInfo(MolName, ChainID)
 324     else:
 325         MiscUtil.PrintError("Failed to retrieve residues information: Selection type %s is not valid..." % SelectionType)
 326 
 327     return SelectionInfo, SelectionLabel

 329 def ListPocketsInfo(MolName, ChainIDs):
 330     """List pockect residues information across all chains."""
 331     
 332     if not (OptionsInfo["PocketLigands"] or OptionsInfo["PocketSolvents"] or OptionsInfo["PocketInorganics"]) :
 333         return
 334     
 335     for ChainID in ChainIDs:
 336         MiscUtil.PrintInfo("\nListing ligand pockets information for chain %s..." % (ChainID))
 337         
 338         LigandsInfo = PyMOLUtil.GetLigandResiduesInfo(MolName, ChainID)
 339         if not len(LigandsInfo["ResNames"]):
 340             MiscUtil.PrintInfo("\nNumber of residues in ligand pocket: None")
 341             MiscUtil.PrintInfo("Chain ID: %s; Ligands: None" % (ChainID))
 342             continue
 343         
 344         for LigandResName in sorted(LigandsInfo["ResNames"]):
 345             for LigandResNum in LigandsInfo["ResNum"][LigandResName]:
 346                 ListPocketsPolymerInfo(MolName, ChainID, LigandResName, LigandResNum)
 347                 ListPocketsSolventsInfo(MolName, ChainID, LigandResName, LigandResNum)
 348                 ListPocketsInorganicsInfo(MolName, ChainID, LigandResName, LigandResNum)

 374 def ListPocketSelectionResiduesInfo(MolName, ChainID, LigandResName, LigandResNum, SelectionType):
 375     """List residues information for a specified pocket selection type. """
 376 
 377     PocketDistanceCutoff  = OptionsInfo["PocketDistanceCutoff"]
 378     SelectionLabel = GetPocketSelectionResiduesInfoLabel(SelectionType)
 379     
 380     SelectionInfo = GetPocketSelectionResiduesInfo(MolName, ChainID, LigandResName, LigandResNum, PocketDistanceCutoff, SelectionType)
 381             
 382     # Setup distribution of residues in the pocket...
 383     LineWords = []
 384     PocketResCount = 0
 385     SortedResNames = sorted(SelectionInfo["ResNames"], key = lambda ResName: SelectionInfo["ResCount"][ResName], reverse = True)
 386     for ResName in SortedResNames:
 387         ResCount = SelectionInfo["ResCount"][ResName]
 388         LineWords.append("%s - %s" % (ResName, ResCount))
 389         PocketResCount += ResCount
 390     
 391     ResidueDistribution = "; ".join(LineWords) if len(LineWords) else None
 392     
 393     # Setup residue IDs sorted by residue numbers...
 394     ResNumMap = {}
 395     for ResName in SelectionInfo["ResNames"]:
 396         for ResNum in SelectionInfo["ResNum"][ResName]:
 397             ResNumMap[ResNum] = ResName
 398     
 399     LineWords = []
 400     for ResNum in sorted(ResNumMap, key = int):
 401         ResName = ResNumMap[ResNum]
 402         ResID = "%s_%s" % (ResName, ResNum)
 403         LineWords.append(ResID)
 404     ResidueIDs = ", ".join(LineWords) if len(LineWords) else None
 405                 
 406     MiscUtil.PrintInfo("\nNumber of %s residues in ligand pocket: %s" % (SelectionLabel, PocketResCount))
 407     MiscUtil.PrintInfo("Chain ID: %s; Ligand ID: %s_%s\nResidue distribution: %s\nResidue IDs: %s" % (ChainID, LigandResName, LigandResNum, ResidueDistribution, ResidueIDs))

 415         SelectionInfo = PyMOLUtil.GetPocketPolymerResiduesInfo(MolName, ChainID, LigandResName, LigandResNum, PocketDistanceCutoff)
 416     elif re.match("^PocketSolvents$", SelectionType, re.I):
 417         SelectionInfo = PyMOLUtil.GetPocketSolventResiduesInfo(MolName, ChainID, LigandResName, LigandResNum, PocketDistanceCutoff)
 418     elif re.match("^PocketInorganics$", SelectionType, re.I):
 419         SelectionInfo = PyMOLUtil.GetPocketInorganicResiduesInfo(MolName, ChainID, LigandResName, LigandResNum, PocketDistanceCutoff)
 420     else:
 421         MiscUtil.PrintError("Failed to retrieve pocket residues information: Selection type %s is not valid..." % SelectionType)
 422 
 423     return SelectionInfo

 425 def ListBoundingBoxInfo(MolName):
 426     """List bounding box information. """
 427     
 428     if not OptionsInfo["BoundingBox"]:
 429         return
 430 
 431     MolSelection = "(%s)" % MolName
 432     MolExtents = pymol.cmd.get_extent(MolSelection)
 433     
 434     XMin, YMin, ZMin = MolExtents[0]
 435     XMax, YMax, ZMax = MolExtents[1]
 436 
 437     XSize = abs(XMax - XMin)
 438     YSize = abs(YMax - YMin)
 439     ZSize = abs(ZMax - ZMin)
 440 
 441     MiscUtil.PrintInfo("\nBounding box coordinates: <XMin, XMax> - <%.3f, %.3f>; <YMin, YMax> - <%.3f, %.3f>; <ZMin, ZMax> - <%.3f, %.3f>" % (XMin, XMax, YMin, YMax, ZMin, ZMax))
 442     MiscUtil.PrintInfo("Bounding box size in angstroms: XSize - %.3f; YSize - %.3f; ZSize - %.3f" % (XSize, YSize, ZSize))
 443

 507 def RetrieveOptions():
