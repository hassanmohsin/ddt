  86 def CalculatePartialCharges():
  87     """Calculate partial atomic charges."""
  88 
  89     Infile = OptionsInfo["Infile"]
  90     Outfile = OptionsInfo["Outfile"]
  91 
  92     MiscUtil.PrintInfo("Calculating partial atomic charges...")
  93     
  94     # Setup a molecule reader...
  95     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
  96     Mols = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
  97     
  98     # Setup a writer...
  99     Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
 100     Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
 101         
 102     if Writer is None:
 103         MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
 104 
 105     MiscUtil.PrintInfo("Generating file %s..." % Outfile)
 106 
 107     # Process molecules...
 108     MolCount, ValidMolCount, CalcFailedCount = [0] * 3 
 109 
 110     for Mol in Mols:
 111         MolCount += 1
 112         if Mol is None:
 113             continue
 114         
 115         if RDKitUtil.IsMolEmpty(Mol):
 116             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 117             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 118             continue
 119         
 120         ValidMolCount += 1
 121         MolWithHs = Chem.AddHs(Mol)
 122 
 123         # Retrieve charges...
 124         PartialCharges = CalculateMolPartialCharges(MolWithHs, MolCount)
 125         if not len(PartialCharges):
 126             CalcFailedCount += 1
 127             continue
 128             
 129         # Write out charges...
 130         WriteMolPartialCharges(Writer, MolWithHs, PartialCharges, Compute2DCoords)
 131         
 132     if Writer is not None:
 133         Writer.close()
 134         
 135     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 136     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 137     MiscUtil.PrintInfo("Number of molecules failed during calculation of partial charges: %d" % CalcFailedCount)
 138     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount + CalcFailedCount))

 140 def CalculateMolPartialCharges(Mol, MolCount):
 141     """Calculate partial atomic charges for a molecule."""
 142     
 143     PartialCharges = []
 144     if OptionsInfo["MMFFChargesMode"]:
 145         if  AllChem.MMFFHasAllMoleculeParams(Mol):
 146             MMFFProp = AllChem.MMFFGetMoleculeProperties(Mol)
 147             PartialCharges = [MMFFProp.GetMMFFPartialCharge(AtomIndex) for AtomIndex in range(Mol.GetNumAtoms())]
 148         else:
 149             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 150             MiscUtil.PrintWarning("Failed to calculate MMFF partial charges for molecule, %s: Missing forcefield parameters" % MolName)
 151             return PartialCharges
 152     else:
 153         rdPartialCharges.ComputeGasteigerCharges(Mol, nIter = OptionsInfo["NumIters"], throwOnParamFailure = OptionsInfo["AllowParamFailure"])
 154         PartialCharges = [Atom.GetProp("_GasteigerCharge") for Atom in Mol.GetAtoms()]
 155     
 156     # Format charges...
 157     PartialCharges = ["%.*f" % (OptionsInfo["Precision"], float(Value)) for Value in PartialCharges]
 158     
 159     return PartialCharges
 160     
 161 def WriteMolPartialCharges(Writer, Mol, PartialCharges, Compute2DCoords):
 162     """Write out partial atomic charges for a molecule."""
 163     
 164     if OptionsInfo["AtomAliasesFormatMode"]:
 165         for Atom, PartialCharge in zip(Mol.GetAtoms(), PartialCharges):
 166             Atom.SetProp('molFileAlias', PartialCharge)
 167     else:
 168         ChargesValues = "\n".join(PartialCharges)
 169         Mol.SetProp(OptionsInfo["DataFieldLabel"], ChargesValues)
 170     
 171     if Compute2DCoords:
 172         AllChem.Compute2DCoords(Mol)
 173     
 174     Writer.write(Mol)

 176 def ProcessOptions():
 177     """Process and validate command line arguments and options"""
 178     
 179     MiscUtil.PrintInfo("Processing options...")
 180     
 181     # Validate options...
 182     ValidateOptions()
 183 
 184     AllowParamFailure = True
 185     if re.match("^No", Options["--allowParamFailure"], re.I):
 186         AllowParamFailure = False
 187     OptionsInfo["AllowParamFailure"] = AllowParamFailure
 188     
 189     AtomAliasesFormatMode = True
 190     if re.match("^DataField", Options["--chargesSDFormat"], re.I):
 191         AtomAliasesFormatMode = False
 192     OptionsInfo["AtomAliasesFormatMode"] = AtomAliasesFormatMode
 193 
 194     OptionsInfo["DataFieldLabel"] = Options["--dataFieldLabel"]
 195     
 196     MMFFChargesMode = False
 197     if re.match("^MMFF", Options["--mode"], re.I):
 198         MMFFChargesMode = True
 199     OptionsInfo["Mode"] = Options["--mode"]
 200     OptionsInfo["MMFFChargesMode"] = MMFFChargesMode
 201     
 202     OptionsInfo["Infile"] = Options["--infile"]
 203     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 204     
 205     OptionsInfo["Outfile"] = Options["--outfile"]
 206     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"], Options["--infile"], Options["--outfile"])
 207     
 208     OptionsInfo["Overwrite"] = Options["--overwrite"]
 209     
 210     OptionsInfo["NumIters"] = int(Options["--numIters"])
 211     OptionsInfo["Precision"] = int(Options["--precision"])
 212 
 213 def RetrieveOptions():
 214     """Retrieve command line arguments and options"""
 215     
 216     # Get options...
 217     global Options
 218     Options = docopt(_docoptUsage_)
 219     
 220     # Set current working directory to the specified directory...
 221     WorkingDir = Options["--workingdir"]
 222     if WorkingDir:
 223         os.chdir(WorkingDir)
 224     
 225     # Handle examples option...
 226     if "--examples" in Options and Options["--examples"]:
 227         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(_docoptUsage_))
 228         sys.exit(0)
 229         
 230 def ValidateOptions():
 231     """Validate option values"""
 232 
 233     MiscUtil.ValidateOptionTextValue("-a, --allowParamFailure", Options["--allowParamFailure"], "yes no")
 234     MiscUtil.ValidateOptionTextValue("-c, --chargesSDFormat", Options["--chargesSDFormat"], "AtomAliases DataField")
 235     
 236     MiscUtil.ValidateOptionTextValue("-m, --mode", Options["--mode"], "Gasteiger MMFF")
 237     
 238     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 239     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd mol smi csv tsv txt")
 240     
 241     MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd")
 242     MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 243     MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 244 
 245     MiscUtil.ValidateOptionIntegerValue("-n, --numIters", Options["--numIters"], {">": 0})
 246     MiscUtil.ValidateOptionIntegerValue("-p, --precision", Options["--precision"], {">": 0})
