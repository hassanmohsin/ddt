  81 def GenerateConformers():
  82     """Generate conformers."""
  83     
  84     Infile = OptionsInfo["Infile"]
  85     Outfile = OptionsInfo["Outfile"]
  86     
  87     # Setup a molecule reader...
  88     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
  89     Mols  = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
  90     
  91     # Set up a molecule writer...
  92     Writer = None
  93     Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
  94     if Writer is None:
  95         MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
  96     MiscUtil.PrintInfo("Generating file %s...\n" % Outfile)
  97 
  98     # Process molecules...
  99     MolCount = 0
 100     ValidMolCount = 0
 101     ConfGenFailedCount = 0
 102 
 103     SkipMinimization = OptionsInfo["SkipForceFieldMinimization"]
 104     
 105     if SkipMinimization:
 106         MiscUtil.PrintInfo("Generating conformers without performing energy minimization...\n")
 107     else:
 108         MiscUtil.PrintInfo("Generating conformers and performing energy minimization...\n")
 109         
 110     for Mol in Mols:
 111         MolCount += 1
 112         
 113         if Mol is None:
 114             continue
 115         
 116         if RDKitUtil.IsMolEmpty(Mol):
 117             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 118             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 119             continue
 120         
 121         ValidMolCount += 1
 122 
 123         if SkipMinimization:
 124             Status = GenerateMolConformers(Mol, MolCount, Writer)
 125         else:
 126             Status = GenerateAndMinimizeMolConformers(Mol, MolCount, Writer)
 127         
 128         if not Status:
 129             ConfGenFailedCount += 1
 130     
 131     if Writer is not None:
 132         Writer.close()
 133     
 134     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 135     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 136     MiscUtil.PrintInfo("Number of molecules failed during conformation generation or minimization: %d" % ConfGenFailedCount)
 137     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount + ConfGenFailedCount))

 139 def GenerateMolConformers(Mol, MolCount, Writer):
 140     "Generate conformers for a molecule and write them out."
 141     
 142     MolName = RDKitUtil.GetMolName(Mol, MolCount)
 143     
 144     ConfIDs = EmbedMolecule(Mol)
 145     if not len(ConfIDs):
 146         MiscUtil.PrintWarning("Conformation generation couldn't be performed for molecule %s: Embedding failed...\n" % MolName)
 147         return False
 148     
 149     if OptionsInfo["AlignConformers"]:
 150         AllChem.AlignMolConformers(Mol)
 151     
 152     
 153     # Write out the conformers...
 154     for ConfID in ConfIDs:
 155         SetConfMolName(Mol, MolName, ConfID)
 156         Writer.write(Mol, confId = ConfID)
 157     
 158     MiscUtil.PrintInfo("\nNumber of conformations written for %s: %d" % (MolName, len(ConfIDs)))
 159     
 160     return True
 161     
 162 def GenerateAndMinimizeMolConformers(Mol, MolCount, Writer):
 163     "Generate and mininize conformers for a molecule and write them out."
 164     
 165     ConfIDs = EmbedMolecule(Mol)
 166     if not len(ConfIDs):
 167         MolName = RDKitUtil.GetMolName(Mol, MolCount)
 168         MiscUtil.PrintWarning("Conformation generation couldn't be performed for molecule %s: Embedding failed...\n" % MolName)
 169         return False
 170 
 171     if  OptionsInfo["AddHydrogens"]:
 172         Mol = Chem.AddHs(Mol)
 173 
 174     CalcEnergyMap = {}
 175     for ConfID in ConfIDs:
 176         try:
 177             if OptionsInfo["UseUFF"]:
 178                 Status = AllChem.UFFOptimizeMolecule(Mol, confId = ConfID, maxIters = OptionsInfo["MaxIters"])
 179             elif OptionsInfo["UseMMFF"]:
 180                 Status = AllChem.MMFFOptimizeMolecule(Mol, confId = ConfID, maxIters = OptionsInfo["MaxIters"])
 181             else:
 182                 MiscUtil.PrintError("Minimization couldn't be performed: Specified forcefield, %s, is not supported" % OptionsInfo["ForceField"])
 183         except RuntimeError as ErrMsg:
 184             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 185             MiscUtil.PrintWarning("Minimization couldn't be performed for molecule %s:\n%s\n" % (MolName, ErrMsg))
 186             return False
 187         
 188         EnergyStatus, Energy = GetConformerEnergy(Mol, ConfID)
 189         if not EnergyStatus:
 190             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 191             MiscUtil.PrintWarning("Failed to retrieve calculated energy for conformation number %d of molecule %s. Try again after removing any salts or cleaing up the molecule...\n" % (ConfID, MolName))
 192             return False
 193         
 194         if Status != 0:
 195             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 196             MiscUtil.PrintWarning("Minimization failed to converge for conformation number %d of molecule %s in %d steps. Try using higher value for \"--maxIters\" option...\n" % (ConfID, MolName, OptionsInfo["MaxIters"]))
 197             
 198         CalcEnergyMap[ConfID] = Energy
 199 
 200     if  OptionsInfo["RemoveHydrogens"]:
 201         Mol = Chem.RemoveHs(Mol)
 202 
 203     # Align molecules after minimization...
 204     if OptionsInfo["AlignConformers"]:
 205         AllChem.AlignMolConformers(Mol)
 206     
 207     SortedConfIDs = sorted(ConfIDs, key = lambda ConfID: CalcEnergyMap[ConfID])
 208     
 209     MinEnergyConfID = SortedConfIDs[0]
 210     MinConfEnergy = CalcEnergyMap[MinEnergyConfID]
 211 
 212     EnergyWindow = OptionsInfo["EnergyWindow"]
 213     MolName = RDKitUtil.GetMolName(Mol, MolCount)
 214 
 215     EnergyRMSDCutoff = OptionsInfo["EnergyRMSDCutoff"]
 216     ApplyEnergyRMSDCutoff = False
 217     if EnergyRMSDCutoff > 0:
 218         ApplyEnergyRMSDCutoff = True
 219 
 220     # Calculate RMSD values for conformers...
 221     PreAligned = False
 222     if OptionsInfo["AlignConformers"]:
 223         PreAligned = True
 224     
 225     CalcRMSDMap = {}
 226     if ApplyEnergyRMSDCutoff:
 227         for ConfID in SortedConfIDs:
 228             RMSD = AllChem.GetConformerRMS(Mol, MinEnergyConfID, ConfID, prealigned=PreAligned)
 229             CalcRMSDMap[ConfID] = RMSD
 230     
 231     # Write out the conformers with in the specified energy window  from the lowest
 232     # energy conformation along with applying RMSD cutoff as needed...
 233     #
 234     ConfCount = 0
 235     IgnoredByEnergyConfCount = 0
 236     IgnoredByRMSDConfCount = 0
 237     
 238     FirstConf = True
 239     
 240     for ConfID in SortedConfIDs:
 241         if FirstConf:
 242             FirstConf = False
 243             # Write it out...
 244             ConfCount += 1
 245             SetConfMolName(Mol, MolName, ConfID)
 246             Writer.write(Mol, confId = ConfID)
 247             continue
 248             
 249         ConfEnergyDiff = abs(CalcEnergyMap[ConfID] - MinConfEnergy )
 250         if  ConfEnergyDiff > EnergyWindow:
 251             IgnoredByEnergyConfCount += 1
 252             continue
 253         
 254         if ApplyEnergyRMSDCutoff:
 255             if CalcRMSDMap[ConfID] < EnergyRMSDCutoff:
 256                 IgnoredByRMSDConfCount += 1
 257                 continue
 258 
 259         # Write it out...
 260         ConfCount += 1
 261         SetConfMolName(Mol, MolName, ConfID)
 262         Writer.write(Mol, confId = ConfID)
 263     
 264     MiscUtil.PrintInfo("\nTotal Number of conformations written for %s: %d" % (MolName, ConfCount))
 265     MiscUtil.PrintInfo("Number of conformations ignored due to energy window cutoff: %d" % (IgnoredByEnergyConfCount))
 266     if ApplyEnergyRMSDCutoff:
 267         MiscUtil.PrintInfo("Number of conformations ignored due to energy RMSD cutoff:  %d" % (IgnoredByRMSDConfCount))
 268     
 269     return True
 270     
 271 def GetConformerEnergy(Mol, ConfID):
 272     "Calculate conformer energy"
 273 
 274     Status = True
 275     Energy = 0.0
 276     
 277     if OptionsInfo["UseUFF"]:
 278         UFFMoleculeForcefield = AllChem.UFFGetMoleculeForceField(Mol, confId = ConfID)
 279         if UFFMoleculeForcefield is None:
 280             Status = False
 281         else:
 282             Energy = UFFMoleculeForcefield.CalcEnergy()
 283     elif OptionsInfo["UseMMFF"]:
 284         MMFFMoleculeProperties = AllChem.MMFFGetMoleculeProperties(Mol)
 285         MMFFMoleculeForcefield = AllChem.MMFFGetMoleculeForceField(Mol, MMFFMoleculeProperties, confId = ConfID)
 286         if MMFFMoleculeForcefield is None:
 287             Status = False
 288         else:
 289             Energy = MMFFMoleculeForcefield.CalcEnergy()
 290     else:
 291         MiscUtil.PrintError("Couldn't retrieve conformer energy: Specified forcefield, %s, is not supported" % OptionsInfo["ForceField"])
 292 
 293     return (Status, Energy)
 294     
 295 def EmbedMolecule(Mol):
 296     "Embed conformations"
 297     
 298     ConfIDs = []
 299     
 300     # Figure out the number of conformations to embded...
 301     if re.match("^Auto$", OptionsInfo["MaxConfs"], re.I):
 302         NumOfRotBonds = Descriptors.NumRotatableBonds(Mol)
 303         if NumOfRotBonds <= 5:
 304             MaxConfs = 100
 305         elif NumOfRotBonds >=6 and NumOfRotBonds <= 10:
 306             MaxConfs = 200
 307         else:
 308             MaxConfs = 300
 309     else:
 310         MaxConfs = int(OptionsInfo["MaxConfs"])
 311         
 312     RandomSeed = OptionsInfo["RandomSeed"]
 313     EnforceChirality = OptionsInfo["EnforceChirality"]
 314     UseExpTorsionAnglePrefs = OptionsInfo["UseExpTorsionAnglePrefs"]
 315     UseBasicKnowledge = OptionsInfo["UseBasicKnowledge"]
 316     EmbedRMSDCutoff = OptionsInfo["EmbedRMSDCutoff"]
 317     
 318     ConfIDs = AllChem.EmbedMultipleConfs(Mol, numConfs = MaxConfs, randomSeed = RandomSeed, pruneRmsThresh = EmbedRMSDCutoff, enforceChirality = EnforceChirality, useExpTorsionAnglePrefs = UseExpTorsionAnglePrefs, useBasicKnowledge = UseBasicKnowledge)
 319 
 320     return ConfIDs

 322 def SetConfMolName(Mol, MolName, ConfCount):
 323     """Set conf mol name"""
 324 
 325     ConfName = "%s_Conf%d" % (MolName, ConfCount)
 326     Mol.SetProp("_Name", ConfName)

 328 def ProcessOptions():
 329     """Process and validate command line arguments and options"""
 330     
 331     MiscUtil.PrintInfo("Processing options...")
 332     
 333     # Validate options...
 334     ValidateOptions()
 335     
 336     OptionsInfo["Infile"] = Options["--infile"]
 337     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 338     
 339     OptionsInfo["Outfile"] = Options["--outfile"]
 340     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"])
 341     
 342     OptionsInfo["Overwrite"] = Options["--overwrite"]
 343 
 344     OptionsInfo["AddHydrogens"] = True
 345     if re.match("^no$", Options["--addHydrogens"], re.I):
 346         OptionsInfo["AddHydrogens"] = False
 347 
 348     OptionsInfo["AlignConformers"] = True
 349     if re.match("^no$", Options["--alignConformers"], re.I):
 350         OptionsInfo["AlignConformers"] = False
 351 
 352     if re.match("^ETDG$", Options["--conformerGenerator"], re.I):
 353         ConformerGenerator = "ETDG"
 354         UseExpTorsionAnglePrefs = True
 355         UseBasicKnowledge = False
 356     elif re.match("^KDG$", Options["--conformerGenerator"], re.I):
 357         ConformerGenerator = "KDG"
 358         UseExpTorsionAnglePrefs = False
 359         UseBasicKnowledge = True
 360     elif re.match("^ETKDG$", Options["--conformerGenerator"], re.I):
 361         ConformerGenerator = "ETKDG"
 362         UseExpTorsionAnglePrefs = True
 363         UseBasicKnowledge = True
 364     elif re.match("^SDG$", Options["--conformerGenerator"], re.I):
 365         ConformerGenerator = "SDG"
 366         UseExpTorsionAnglePrefs = False
 367         UseBasicKnowledge = False
 368     
 369     OptionsInfo["ConformerGenerator"] = ConformerGenerator
 370     OptionsInfo["UseExpTorsionAnglePrefs"] = UseExpTorsionAnglePrefs
 371     OptionsInfo["UseBasicKnowledge"] = UseBasicKnowledge
 372 
 373     if re.match("^UFF$", Options["--forceField"], re.I):
 374         ForceField = "UFF"
 375         UseUFF = True
 376         UseMMFF = False
 377         SkipForceFieldMinimization = False
 378     elif re.match("^MMFF$", Options["--forceField"], re.I):
 379         ForceField = "MMFF"
 380         UseUFF = False
 381         UseMMFF = True
 382         SkipForceFieldMinimization = False
 383     else:
 384         ForceField = "None"
 385         UseUFF = False
 386         UseMMFF = False
 387         SkipForceFieldMinimization = True
 388         
 389     OptionsInfo["SkipForceFieldMinimization"] = SkipForceFieldMinimization
 390     OptionsInfo["ForceField"] = ForceField
 391     OptionsInfo["UseMMFF"] = UseMMFF
 392     OptionsInfo["UseUFF"] = UseUFF
 393         
 394     OptionsInfo["EnforceChirality"] = True
 395     if re.match("^no$", Options["--enforceChirality"], re.I):
 396         OptionsInfo["EnforceChirality"] = False
 397     
 398     OptionsInfo["EnergyWindow"] = float(Options["--energyWindow"])
 399     
 400     EmbedRMSDCutoff = -1.0
 401     if not re.match("^none$", Options["--embedRMSDCutoff"], re.I):
 402         EmbedRMSDCutoff = float(Options["--embedRMSDCutoff"])
 403     OptionsInfo["EmbedRMSDCutoff"] = EmbedRMSDCutoff
 404     
 405     EnergyRMSDCutoff = -1.0
 406     if not re.match("^none$", Options["--energyRMSDCutoff"], re.I):
 407         EnergyRMSDCutoff = float(Options["--energyRMSDCutoff"])
 408     OptionsInfo["EnergyRMSDCutoff"] = EnergyRMSDCutoff
 409     
 410     OptionsInfo["MaxIters"] = int(Options["--maxIters"])
 411     OptionsInfo["MaxConfs"] = Options["--maxConfs"]
 412     
 413     RandomSeed = -1
 414     if not re.match("^auto$", Options["--randomSeed"], re.I):
 415         RandomSeed = int(Options["--randomSeed"])
 416     OptionsInfo["RandomSeed"] = RandomSeed
 417     
 418     OptionsInfo["RemoveHydrogens"] = True
 419     if re.match("^no$", Options["--removeHydrogens"], re.I):
 420         OptionsInfo["RemoveHydrogens"] = False
 421     
 422 def RetrieveOptions():
 423     """Retrieve command line arguments and options"""
 424     
 425     # Get options...
 426     global Options
 427     Options = docopt(_docoptUsage_)
 428 
 429     # Set current working directory to the specified directory...
 430     WorkingDir = Options["--workingdir"]
 431     if WorkingDir:
 432         os.chdir(WorkingDir)
 433     
 434     # Handle examples option...
 435     if "--examples" in Options and Options["--examples"]:
 436         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(_docoptUsage_))
 437         sys.exit(0)

 439 def ValidateOptions():
 440     """Validate option values"""
 441     
 442     MiscUtil.ValidateOptionTextValue("-a, --addHydrogens", Options["--addHydrogens"], "yes no")
 443     MiscUtil.ValidateOptionTextValue("--alignConformers", Options["--alignConformers"], "yes no")
 444     MiscUtil.ValidateOptionTextValue("-c, --conformerGenerator", Options["--conformerGenerator"], "SDG ETDG KDG ETKDG")
 445     MiscUtil.ValidateOptionTextValue("-f, --forceField", Options["--forceField"], "UFF MMFF None")
 446     
 447     MiscUtil.ValidateOptionTextValue("--enforceChirality ", Options["--enforceChirality"], "yes no")
 448     MiscUtil.ValidateOptionFloatValue("--energyWindow", Options["--energyWindow"], {">": 0.0})
 449     
 450     if not re.match("^none$", Options["--embedRMSDCutoff"], re.I):
 451         MiscUtil.ValidateOptionFloatValue("--embedRMSDCutoff", Options["--embedRMSDCutoff"], {">": 0.0})
 452     
 453     if not re.match("^none$", Options["--energyRMSDCutoff"], re.I):
 454         MiscUtil.ValidateOptionFloatValue("--energyRMSDCutoff", Options["--energyRMSDCutoff"], {">": 0.0})
 455         # Make sure that the alignConformers option is being used...
 456         if not re.match("^yes$", Options["--alignConformers"], re.I):
 457             MiscUtil.PrintError("%s value of \"--alignConformers\" is not allowed for %s value of \"--energyRMSDCutoff\" option " % (Options["--alignConformers"], Options["--energyRMSDCutoff"]))
 458         
 459     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 460     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd mol smi txt csv tsv")
 461     
 462     MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd")
 463     MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 464     MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 465         
 466     if not re.match("^auto$", Options["--maxConfs"], re.I):
 467         MiscUtil.ValidateOptionIntegerValue("--maxConfs", Options["--maxConfs"], {">": 0})
 468     
 469     MiscUtil.ValidateOptionIntegerValue("--maxIters", Options["--maxIters"], {">": 0})
 470     
 471     if not re.match("^auto$", Options["--randomSeed"], re.I):
 472         MiscUtil.ValidateOptionIntegerValue("--randomSeed", Options["--randomSeed"], {})
 473         RandomSeed = int(Options["--randomSeed"])
 474     
 475     MiscUtil.ValidateOptionTextValue("-r, --removeHydrogens", Options["--removeHydrogens"], "yes no")
