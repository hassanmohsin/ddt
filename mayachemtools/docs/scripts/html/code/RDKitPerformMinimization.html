  81 def PerformMinimization():
  82     """Perform minimization."""
  83     
  84     Infile = OptionsInfo["Infile"]
  85     Outfile = OptionsInfo["Outfile"]
  86     
  87     # Setup a molecule reader...
  88     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
  89     Mols  = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
  90     
  91     # Set up a molecule writer...
  92     Writer = None
  93     Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
  94     if Writer is None:
  95         MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
  96     MiscUtil.PrintInfo("Generating file %s...\n" % Outfile)
  97 
  98     # Process molecules...
  99     MolCount = 0
 100     ValidMolCount = 0
 101     MinimizationFailedCount = 0
 102 
 103     if OptionsInfo["SkipConformerGeneration"]:
 104         MiscUtil.PrintInfo("Performing minimization without generation of conformers...\n")
 105     else:
 106         MiscUtil.PrintInfo("Performing minimization with generation of conformers...\n")
 107         
 108     for Mol in Mols:
 109         MolCount += 1
 110         
 111         if Mol is None:
 112             continue
 113         
 114         if RDKitUtil.IsMolEmpty(Mol):
 115             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 116             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 117             continue
 118         
 119         ValidMolCount += 1
 120 
 121         if OptionsInfo["SkipConformerGeneration"]:
 122             Status = MinimizeMolecule(Mol, MolCount, Writer)
 123         else:
 124             Status = GenerateAndMinimizeConformers(Mol, MolCount, Writer)
 125         
 126         if not Status:
 127             MinimizationFailedCount += 1
 128             
 129     if Writer is not None:
 130         Writer.close()
 131     
 132     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 133     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 134     MiscUtil.PrintInfo("Number of molecules failed during conformation generation or minimization: %d" % MinimizationFailedCount)
 135     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount + MinimizationFailedCount))
 136 
 137 def MinimizeMolecule(Mol, MolCount, Writer):
 138     "Minimize moleculer and write it out"
 139     
 140     if  OptionsInfo["AddHydrogens"]:
 141         Mol = Chem.AddHs(Mol)
 142 
 143     Status = 0
 144     try:
 145         if OptionsInfo["UseUFF"]:
 146             Status = AllChem.UFFOptimizeMolecule(Mol, maxIters = OptionsInfo["MaxIters"])
 147         elif OptionsInfo["UseMMFF"]:
 148             Status = AllChem.MMFFOptimizeMolecule(Mol,  maxIters = OptionsInfo["MaxIters"])
 149         else:
 150             MiscUtil.PrintError("Minimization couldn't be performed: Specified forcefield, %s, is not supported" % OptionsInfo["ForceField"])
 151     except RuntimeError as ErrMsg:
 152         MolName = RDKitUtil.GetMolName(Mol, MolCount)
 153         MiscUtil.PrintWarning("Minimization couldn't be performed for molecule %s:\n%s\n" % (MolName, ErrMsg))
 154         return False
 155     
 156     if Status != 0:
 157         MolName = RDKitUtil.GetMolName(Mol, MolCount)
 158         MiscUtil.PrintWarning("Minimization failed to converge for molecule %s in %d steps. Try using higher value for \"--maxIters\" option...\n" % (MolName, OptionsInfo["MaxIters"]))
 159         
 160     if  OptionsInfo["RemoveHydrogens"]:
 161         Mol = Chem.RemoveHs(Mol)
 162     
 163     Writer.write(Mol)
 164     
 165     return True
 166 
 167 def GenerateAndMinimizeConformers(Mol, MolCount, Writer):
 168     "Generate and mininize conformers for a molecule and write out the lowest energy conformer."
 169     
 170     ConfIDs = EmbedMolecule(Mol)
 171     if not len(ConfIDs):
 172         MolName = RDKitUtil.GetMolName(Mol, MolCount)
 173         MiscUtil.PrintWarning("Minimization couldn't be performed for molecule %s: Embedding failed...\n" % MolName)
 174         return False
 175 
 176     CalcEnergyMap = {}
 177     for ConfID in ConfIDs:
 178         try:
 179             if OptionsInfo["UseUFF"]:
 180                 Status = AllChem.UFFOptimizeMolecule(Mol, confId = ConfID, maxIters = OptionsInfo["MaxIters"])
 181             elif OptionsInfo["UseMMFF"]:
 182                 Status = AllChem.MMFFOptimizeMolecule(Mol, confId = ConfID, maxIters = OptionsInfo["MaxIters"])
 183             else:
 184                 MiscUtil.PrintError("Minimization couldn't be performed: Specified forcefield, %s, is not supported" % OptionsInfo["ForceField"])
 185         except RuntimeError as ErrMsg:
 186             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 187             MiscUtil.PrintWarning("Minimization couldn't be performed for molecule %s:\n%s\n" % (MolName, ErrMsg))
 188             return False
 189         
 190         EnergyStatus, Energy = GetConformerEnergy(Mol, ConfID)
 191         if not EnergyStatus:
 192             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 193             MiscUtil.PrintWarning("Failed to retrieve calculated energy for conformation number %d of molecule %s. Try again after removing any salts or cleaing up the molecule...\n" % (ConfID, MolName))
 194             return False
 195         
 196         if Status != 0:
 197             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 198             MiscUtil.PrintWarning("Minimization failed to converge for conformation number %d of molecule %s in %d steps. Try using higher value for \"--maxIters\" option...\n" % (ConfID, MolName, OptionsInfo["MaxIters"]))
 199             
 200         CalcEnergyMap[ConfID] = Energy
 201 
 202     SortedConfIDs = sorted(ConfIDs, key = lambda ConfID: CalcEnergyMap[ConfID])
 203     MinEnergyConfID = SortedConfIDs[0]
 204         
 205     if  OptionsInfo["RemoveHydrogens"]:
 206         Mol = Chem.RemoveHs(Mol)
 207     
 208     Writer.write(Mol, confId = MinEnergyConfID)
 209     
 210     return True
 211     
 212 def GetConformerEnergy(Mol, ConfID):
 213     "Calculate conformer energy"
 214 
 215     Status = True
 216     Energy = 0.0
 217     
 218     if OptionsInfo["UseUFF"]:
 219         UFFMoleculeForcefield = AllChem.UFFGetMoleculeForceField(Mol, confId = ConfID)
 220         if UFFMoleculeForcefield is None:
 221             Status = False
 222         else:
 223             Energy = UFFMoleculeForcefield.CalcEnergy()
 224     elif OptionsInfo["UseMMFF"]:
 225         MMFFMoleculeProperties = AllChem.MMFFGetMoleculeProperties(Mol)
 226         MMFFMoleculeForcefield = AllChem.MMFFGetMoleculeForceField(Mol, MMFFMoleculeProperties, confId = ConfID)
 227         if MMFFMoleculeForcefield is None:
 228             Status = False
 229         else:
 230             Energy = MMFFMoleculeForcefield.CalcEnergy()
 231     else:
 232         MiscUtil.PrintError("Couldn't retrieve conformer energy: Specified forcefield, %s, is not supported" % OptionsInfo["ForceField"])
 233     
 234     return (Status, Energy)
 235     
 236 def EmbedMolecule(Mol):
 237     "Embed conformations"
 238     
 239     ConfIDs = []
 240     
 241     MaxConfs = OptionsInfo["MaxConfs"]
 242     RandomSeed = OptionsInfo["RandomSeed"]
 243     EnforceChirality = OptionsInfo["EnforceChirality"]
 244     UseExpTorsionAnglePrefs = OptionsInfo["UseExpTorsionAnglePrefs"]
 245     UseBasicKnowledge = OptionsInfo["UseBasicKnowledge"]
 246 
 247     ConfIDs = AllChem.EmbedMultipleConfs(Mol, numConfs = MaxConfs, randomSeed = RandomSeed, enforceChirality = EnforceChirality, useExpTorsionAnglePrefs = UseExpTorsionAnglePrefs, useBasicKnowledge = UseBasicKnowledge)
 248     
 249     return ConfIDs

 251 def ProcessOptions():
 252     """Process and validate command line arguments and options"""
 253     
 254     MiscUtil.PrintInfo("Processing options...")
 255     
 256     # Validate options...
 257     ValidateOptions()
 258     
 259     OptionsInfo["Infile"] = Options["--infile"]
 260     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 261     
 262     OptionsInfo["Outfile"] = Options["--outfile"]
 263     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"])
 264     
 265     OptionsInfo["Overwrite"] = Options["--overwrite"]
 266 
 267     OptionsInfo["AddHydrogens"] = True
 268     if re.match("^no$", Options["--addHydrogens"], re.I):
 269         OptionsInfo["AddHydrogens"] = False
 270 
 271     if re.match("^ETDG$", Options["--conformerGenerator"], re.I):
 272         ConformerGenerator = "ETDG"
 273         UseExpTorsionAnglePrefs = True
 274         UseBasicKnowledge = False
 275         SkipConformerGeneration = False
 276     elif re.match("^KDG$", Options["--conformerGenerator"], re.I):
 277         ConformerGenerator = "KDG"
 278         UseExpTorsionAnglePrefs = False
 279         UseBasicKnowledge = True
 280         SkipConformerGeneration = False
 281     elif re.match("^ETKDG$", Options["--conformerGenerator"], re.I):
 282         ConformerGenerator = "ETKDG"
 283         UseExpTorsionAnglePrefs = True
 284         UseBasicKnowledge = True
 285         SkipConformerGeneration = False
 286     elif re.match("^SDG$", Options["--conformerGenerator"], re.I):
 287         ConformerGenerator = "SDG"
 288         UseExpTorsionAnglePrefs = False
 289         UseBasicKnowledge = False
 290         SkipConformerGeneration = False
 291     else:
 292         ConformerGenerator = "None"
 293         UseExpTorsionAnglePrefs = False
 294         UseBasicKnowledge = False
 295         SkipConformerGeneration = True
 296     
 297     OptionsInfo["SkipConformerGeneration"] = SkipConformerGeneration
 298     OptionsInfo["ConformerGenerator"] = ConformerGenerator
 299     OptionsInfo["UseExpTorsionAnglePrefs"] = UseExpTorsionAnglePrefs
 300     OptionsInfo["UseBasicKnowledge"] = UseBasicKnowledge
 301 
 302     if re.match("^UFF$", Options["--forceField"], re.I):
 303         ForceField = "UFF"
 304         UseUFF = True
 305         UseMMFF = False
 306     elif re.match("^MMFF$", Options["--forceField"], re.I):
 307         ForceField = "MMFF"
 308         UseUFF = False
 309         UseMMFF = True
 310     
 311     OptionsInfo["ForceField"] = ForceField
 312     OptionsInfo["UseMMFF"] = UseMMFF
 313     OptionsInfo["UseUFF"] = UseUFF
 314         
 315     OptionsInfo["EnforceChirality"] = True
 316     if re.match("^no$", Options["--enforceChirality"], re.I):
 317         OptionsInfo["EnforceChirality"] = False
 318     
 319     OptionsInfo["MaxIters"] = int(Options["--maxIters"])
 320     OptionsInfo["MaxConfs"] = int(Options["--maxConfs"])
 321     
 322     RandomSeed = -1
 323     if not re.match("^auto$", Options["--randomSeed"], re.I):
 324         RandomSeed = int(Options["--randomSeed"])
 325     OptionsInfo["RandomSeed"] = RandomSeed
 326     
 327     OptionsInfo["RemoveHydrogens"] = True
 328     if re.match("^no$", Options["--removeHydrogens"], re.I):
 329         OptionsInfo["RemoveHydrogens"] = False
 330 
 331 def RetrieveOptions():
 332     """Retrieve command line arguments and options"""
 333     
 334     # Get options...
 335     global Options
 336     Options = docopt(_docoptUsage_)
 337     
 338     # Set current working directory to the specified directory...
 339     WorkingDir = Options["--workingdir"]
 340     if WorkingDir:
 341         os.chdir(WorkingDir)
 342     
 343     # Handle examples option...
 344     if "--examples" in Options and Options["--examples"]:
 345         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(_docoptUsage_))
 346         sys.exit(0)

 348 def ValidateOptions():
 349     """Validate option values"""
 350     
 351     MiscUtil.ValidateOptionTextValue("-a, --addHydrogens", Options["--addHydrogens"], "yes no")
 352     MiscUtil.ValidateOptionTextValue("-c, --conformerGenerator", Options["--conformerGenerator"], "SDG ETDG KDG ETKDG None")
 353     MiscUtil.ValidateOptionTextValue("-f, --forceField", Options["--forceField"], "UFF MMFF")
 354     
 355     MiscUtil.ValidateOptionTextValue("--enforceChirality ", Options["--enforceChirality"], "yes no")
 356     
 357     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 358     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd mol smi txt csv tsv")
 359     
 360     MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd")
 361     MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 362     MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 363         
 364     MiscUtil.ValidateOptionIntegerValue("--maxConfs", Options["--maxConfs"], {">": 0})
 365     MiscUtil.ValidateOptionIntegerValue("--maxIters", Options["--maxIters"], {">": 0})
 366     
 367     if not re.match("^auto$", Options["--randomSeed"], re.I):
 368         MiscUtil.ValidateOptionIntegerValue("--randomSeed", Options["--randomSeed"], {})
 369         RandomSeed = int(Options["--randomSeed"])
 370     
 371     MiscUtil.ValidateOptionTextValue("-r, --removeHydrogens", Options["--removeHydrogens"], "yes no")

 373 # Setup a usage string for docopt...
 374 _docoptUsage_ = """
 375 RDKitPerformMinimization.py - Perform structure minimization
 376 
 377 Usage:
 378     RDKitPerformMinimization.py [--addHydrogens <yes or no>] [--conformerGenerator <SDG, ETDG, KDG, ETKDG, None> ]
 379                                 [--forceField <UFF or MMFF>] [--enforceChirality <yes or no>] [--infileParams <Name,Value,...>]
 380                                 [--maxConfs <number>] [--maxIters <number>] [ --outfileParams <Name,Value,...> ]
 381                                 [--overwrite] [ --removeHydrogens <yes or no>] [--randomSeed <number>] [-w <dir>] -i <infile> -o <outfile> 
 382     RDKitPerformMinimization.py -h | --help | -e | --examples
 383

 385     Generate 3D structures for molecules using combination of distance geometry
 386     and forcefield minimization or minimize existing 3D structures using a specified
 387     forcefield.
 388 
 389     The supported input file formats are: Mol (.mol), SD (.sdf, .sd), SMILES (.smi)
 390     .csv, .tcsv .txt)
 391 
 392     The supported output file formats are: SD (.sdf, .sd)
 393 
 394 Options:
 395     -a, --addHydrogens <yes or no>  [default: yes]
 396         Add hydrogens before minimization.
 397     -c, --conformerGenerator <SDG, ETDG, KDG, ETKDG, None>  [default: ETKDG]
 398         Conformation generation methodology for generating initial 3D coordinates.
 399         Possible values: Standard Distance Geometry, (SDG), Experimental Torsion-angle
 400         preference with Distance Geometry (ETDG), basic Knowledge-terms with Distance
 401         Geometry (KDG),  and Experimental Torsion-angle preference along with basic
 402         Knowledge-terms with Distance Geometry (ETKDG) [Ref 129] .
 403         
 404         The conformation generation step may be skipped by specifying 'None' value to
 405         perform only forcefield minimization of molecules with 3D structures in input
 406         file.  This doesn't work for molecules in SMILES file or molecules in SD/MOL files
 407         containing 2D structures.
 408     -f, --forceField <UFF or MMFF>  [default: MMFF]
 409         Forcefield method to use for energy minimization. Possible values: Universal Force
 410         Field (UFF) [ Ref 81 ] or Merck Molecular Mechanics Force Field (MMFF) [ Ref 83-87 ] .
 411     --enforceChirality <yes or no>  [default: Yes]
 412         Enforce chirality for defined chiral centers.
 413     -e, --examples
 414         Print examples.
 415     -h, --help
 416         Print this help message.
 417     -i, --infile <infile>
 418         Input file name.
 419     --infileParams <Name,Value,...>  [default: auto]
 420         A comma delimited list of parameter name and value pairs for reading
 421         molecules from files. The supported parameter names for different file
 422         formats, along with their default values, are shown below:
 423             
 424             SD, MOL: removeHydrogens,yes,sanitize,yes,strictParsing,yes
 425             SMILES: smilesColumn,1,smilesNameColumn,2,smilesDelimiter,space,
 426                 smilesTitleLine,auto,sanitize,yes
 427             
 428         Possible values for smilesDelimiter: space, comma or tab.
 429     --maxConfs <number>  [default: 250]
 430         Maximum number of conformations to generate for each molecule by conformation
 431         generation methodology for initial 3D coordinates. The conformations are minimized
 432         using the specified forcefield and the lowest energy conformation is written to the
 433         output file. This option is ignored during 'None' value of '-c --conformerGenerator'
 434         option.

 437         minimization.
 438     -o, --outfile <outfile>
 439         Output file name.
 440     --outfileParams <Name,Value,...>  [default: auto]
 441         A comma delimited list of parameter name and value pairs for writing

 451     --randomSeed <number>  [default: auto]

 454     -w, --workingdir <dir>
