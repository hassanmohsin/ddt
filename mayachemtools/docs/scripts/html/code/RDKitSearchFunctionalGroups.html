  89 def PerformSearch():
  90     """Perform search using SMARTS pattern for specified functional groups."""
  91 
  92     Infile = OptionsInfo["Infile"]
  93     Outfile = OptionsInfo["Outfile"]
  94 
  95     Groups = OptionsInfo["SpecifiedFunctionalGroups"]
  96     GroupsNegateMatch = OptionsInfo["SpecifiedFunctionalGroupsNegateMatch"]
  97     GroupsPatternMols = OptionsInfo["SpecifiedFunctionalGroupsPatternMols"]
  98     
  99     GroupsOutfiles = OptionsInfo["SpecifiedFunctionalGroupsOutfiles"]
 100     GroupsCount = len(Groups)
 101 
 102     CombineMatchResults = OptionsInfo["CombineMatchResults"]
 103     AndCombineOperatorMode = OptionsInfo["AndCombineOperatorMode"]
 104     
 105     CountMode = OptionsInfo["CountMode"]
 106     UseChirality = OptionsInfo["UseChirality"]
 107 
 108     # Setup a molecule reader...
 109     MiscUtil.PrintInfo("\nProcessing file %s..." % Infile)
 110     Mols  = RDKitUtil.ReadMolecules(Infile, **OptionsInfo["InfileParams"])
 111     
 112     # Set up  molecule writers...
 113     Writer = None
 114     GroupOutfilesWriters = []
 115     if not CountMode:
 116         Writer,  GroupOutfilesWriters = SetupMoleculeWriters(CombineMatchResults, Outfile, GroupsOutfiles)
 117     
 118     # Initialize pattern mols match count and status...
 119     GroupsPatternMolsMatchCount = [0] * GroupsCount
 120     GroupsPatternMolsMatchStatus = [False] * GroupsCount
 121 
 122     # Process molecules...
 123     MolCount = 0
 124     ValidMolCount = 0
 125     MatchCount = 0
 126 
 127     Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
 128     
 129     for Mol in Mols:
 130         MolCount += 1
 131         
 132         if Mol is None:
 133             continue
 134         
 135         if RDKitUtil.IsMolEmpty(Mol):
 136             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 137             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 138             continue
 139         
 140         ValidMolCount += 1
 141 
 142         # Match pattern mols...
 143         for GroupIndex in range(0, GroupsCount):
 144             GroupsPatternMolsMatchStatus[GroupIndex] = DoesPatternMolMatch(GroupsPatternMols[GroupIndex], Mol, UseChirality, GroupsNegateMatch[GroupIndex])
 145             if GroupsPatternMolsMatchStatus[GroupIndex]:
 146                 GroupsPatternMolsMatchCount[GroupIndex] += 1
 147         
 148         # Match mol against all specified criteria...
 149         MolMatched = DoesMolMeetSpecifiedMatchCriteria(GroupsPatternMolsMatchStatus, CombineMatchResults, AndCombineOperatorMode)
 150         if MolMatched:
 151             MatchCount += 1
 152             
 153         # Nothing to write...
 154         if CountMode or (not MolMatched):
 155             continue
 156         
 157         # Write out matched molecules...
 158         if Compute2DCoords:
 159             AllChem.Compute2DCoords(Mol)
 160         
 161         if CombineMatchResults:
 162             Writer.write(Mol)
 163         else:
 164             for GroupIndex in range(0, GroupsCount):
 165                 if GroupsPatternMolsMatchStatus[GroupIndex]:
 166                     GroupOutfilesWriters[GroupIndex].write(Mol)
 167     
 168     if Writer is not None:
 169         Writer.close()
 170     for GroupOutfileWriter in GroupOutfilesWriters:
 171         GroupOutfileWriter.close()
 172         
 173     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 174     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 175     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount))
 176 
 177     MiscUtil.PrintInfo("\nTotal number of molecules matched against specified match criteria: %d" % MatchCount)
 178     
 179     MiscUtil.PrintInfo("\nNumber of molecuels matched against individual functional groups:")
 180     MiscUtil.PrintInfo("FunctionalGroupName,MatchCount")
 181     
 182     for GroupIndex in range(0, GroupsCount):
 183         GroupName = Groups[GroupIndex]
 184         if GroupsNegateMatch[GroupIndex]:
 185             GroupName = '!' + GroupName
 186         GroupMatchCount = GroupsPatternMolsMatchCount[GroupIndex]
 187         MiscUtil.PrintInfo("%s,%d" % (GroupName, GroupMatchCount))
 188 
 189 def DoesMolMeetSpecifiedMatchCriteria(GroupsPatternMolsMatchStatus,  CombineMatchResults, AndCombineOperatorMode):
 190     """Match molecule using specified match criteia."""
 191 
 192     if CombineMatchResults and AndCombineOperatorMode:
 193         # Must match all specified SMARTS
 194         Status = True
 195         for MatchStatus in GroupsPatternMolsMatchStatus:
 196             if not MatchStatus:
 197                 Status = False
 198                 break
 199     else:
 200         # One match is enough...
 201         Status = False
 202         for MatchStatus in GroupsPatternMolsMatchStatus:
 203             if MatchStatus:
 204                 Status = True
 205                 break
 206     
 207     return Status
 208     
 209 def SetupMoleculeWriters(CombineMatchResults, Outfile, GroupsOutfiles):
 210     """Set up molecule writers for output files."""
 211 
 212     Writer = None
 213     GroupOutfilesWriters = []
 214     
 215     if CombineMatchResults:
 216         Writer = RDKitUtil.MoleculesWriter(Outfile, **OptionsInfo["OutfileParams"])
 217         if Writer is None:
 218             MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Outfile)
 219         MiscUtil.PrintInfo("Generating file %s..." % Outfile)
 220     else:
 221         for GroupOutfile in GroupsOutfiles:
 222             GroupOutfileWriter = RDKitUtil.MoleculesWriter(GroupOutfile, **OptionsInfo["OutfileParams"])
 223             if GroupOutfileWriter is None:
 224                 MiscUtil.PrintError("Failed to setup a writer for output fie %s " % Writer)
 225             GroupOutfilesWriters.append(GroupOutfileWriter)
 226         
 227         GroupsCount = len(GroupsOutfiles)
 228         if GroupsCount > 4:
 229             MiscUtil.PrintInfo("Generating %d output files with the following file name format: %s<GroupName>.%s" % (GroupsCount, OptionsInfo["OutfileBasename"], OptionsInfo["OutfileExt"]))
 230         else:
 231             Delmiter = ', '
 232             OutfileNames = Delmiter.join(GroupsOutfiles)
 233             MiscUtil.PrintInfo("Generating %d output files: %s..." % (GroupsCount, OutfileNames))
 234 
 235     return (Writer, GroupOutfilesWriters)
 236     
 237 def DoesPatternMolMatch(PatternMol, Mol, UseChirality, NegateMatch):
 238     """Perform a substructure match for the presence of pattern molecule in a molecule."""
 239 
 240     MolMatched = Mol.HasSubstructMatch(PatternMol, useChirality = UseChirality)
 241     if NegateMatch:
 242         if MolMatched:
 243             MolMatched = False
 244         else:
 245             MolMatched = True
 246     
 247     return MolMatched

 249 def ProcessFunctionalGroupsInfo():
 250     """Process functional groups information."""
 251     
 252     RetrieveFunctionalGroupsInfo()
 253     ProcessSpecifiedFunctionalGroups()
 254 
 255     SetupFunctionalGroupsSMARTSPatterns()
 256     SetupFunctionalGroupsOutputFileNames()

 258 def ProcessSpecifiedFunctionalGroups():
 259     """Process and validate specified functional groups"""
 260     
 261     OptionsInfo["SpecifiedFunctionalGroups"] = []
 262     OptionsInfo["SpecifiedFunctionalGroupsNegateMatch"] = []
 263     
 264     if re.match("^All$", OptionsInfo["FunctionalGroups"], re.I):
 265         OptionsInfo["SpecifiedFunctionalGroups"] = FunctionalGroupsMap['Names']
 266         OptionsInfo["SpecifiedFunctionalGroupsNegateMatch"] = [False] * len(OptionsInfo["SpecifiedFunctionalGroups"])
 267         return
 268     
 269     # Set up a map of valid group names for checking specified group names...
 270     CanonicalGroupNameMap = {}
 271     for GroupName in FunctionalGroupsMap['Names']:
 272         CanonicalGroupNameMap[GroupName.lower()] = GroupName
 273         
 274     # Parse and validate specified names...
 275     GroupNames = re.sub(" ", "", OptionsInfo["FunctionalGroups"])
 276     if not GroupNames:
 277         MiscUtil.PrintError("No functional group name specified for \"-f, --functionalGroups\" option")
 278     
 279     SpecifiedFunctionalGroups = []
 280     SpecifiedNegateMatchStatus = []
 281     
 282     for GroupName in GroupNames.split(","):
 283         CanonicalGroupName = GroupName.lower()
 284         NegateMatchStatus = False
 285         if re.match("^!", CanonicalGroupName, re.I):
 286             NegateMatchStatus = True
 287             CanonicalGroupName = re.sub("^!", "", CanonicalGroupName)
 288         if CanonicalGroupName in CanonicalGroupNameMap:
 289             SpecifiedFunctionalGroups.append(CanonicalGroupNameMap[CanonicalGroupName])
 290             SpecifiedNegateMatchStatus.append(NegateMatchStatus)
 291         else:
 292             MiscUtil.PrintWarning("The functional group name, %s, specified using \"-f, --functionalGroups\" option is not a valid name." % (GroupName))
 293 
 294     if not len(SpecifiedFunctionalGroups):
 295         MiscUtil.PrintError("No valid functional group names specified for \"-f, --functionalGroups\" option")
 296         
 297     OptionsInfo["SpecifiedFunctionalGroups"] = SpecifiedFunctionalGroups
 298     OptionsInfo["SpecifiedFunctionalGroupsNegateMatch"] = SpecifiedNegateMatchStatus
 299 
 300 def SetupFunctionalGroupsSMARTSPatterns():
 301     """Setup SMARTS patterns for specified functional groups."""
 302 
 303     OptionsInfo["SpecifiedFunctionalGroupsSMARTSPatterns"] = []
 304     OptionsInfo["SpecifiedFunctionalGroupsPatternMols"] = []
 305     
 306     for Name in OptionsInfo["SpecifiedFunctionalGroups"]:
 307         SMARTSPattern = FunctionalGroupsMap['SMARTSPattern'][Name]
 308         PatternMol = Chem.MolFromSmarts(SMARTSPattern)
 309         if PatternMol is None:
 310             MiscUtil.PrintError("Failed to parse SMARTS pattern, %s, for function group, %s" % (SMARTSPattern, Name))
 311         
 312         OptionsInfo["SpecifiedFunctionalGroupsSMARTSPatterns"].append(SMARTSPattern)
 313         OptionsInfo["SpecifiedFunctionalGroupsPatternMols"].append(PatternMol)
 314 
 315 def SetupFunctionalGroupsOutputFileNames():
 316     """Setup output file names for specified functional group names."""
 317 
 318     OptionsInfo["SpecifiedFunctionalGroupsOutfiles"] = []
 319     
 320     if OptionsInfo["CountMode"]:
 321         # No need of any output file...
 322         return
 323     
 324     if OptionsInfo["CombineMatchResults"]:
 325         # No need of output files for specified functional groups...
 326         return
 327     
 328     OutfileBasename = OptionsInfo["OutfileBasename"]
 329     OutfileExt = OptionsInfo["OutfileExt"]
 330     SpecifiedFunctionalGroupsOutfiles = []
 331 
 332     GroupsCount = len(OptionsInfo["SpecifiedFunctionalGroups"])
 333     for GroupIndex in range(0, GroupsCount):
 334         GroupName = OptionsInfo["SpecifiedFunctionalGroups"][GroupIndex]
 335         if OptionsInfo["SpecifiedFunctionalGroupsNegateMatch"][GroupIndex]:
 336             GroupName = "Not" + GroupName
 337         GroupName = re.sub("\.", "", GroupName)
 338         
 339         GroupOutfile = "%s%s.%s" % (OutfileBasename, GroupName, OutfileExt)
 340         SpecifiedFunctionalGroupsOutfiles.append(GroupOutfile)
 341 
 342     OptionsInfo["SpecifiedFunctionalGroupsOutfiles"] = SpecifiedFunctionalGroupsOutfiles
 343     
 344 def RetrieveFunctionalGroupsInfo():
 345     """Retrieve functional groups information"""
 346 
 347     MiscUtil.PrintInfo("\nRetrieving data from default RDKit functional groups hierarchy file Functional_Group_Hierarchy.txt...")
 348     
 349     FunctionalGroupNamesFile = OptionsInfo["GroupNamesFile"]
 350     FunctionalGroupsNodes = FunctionalGroups.BuildFuncGroupHierarchy(FunctionalGroupNamesFile)
 351 
 352     FunctionalGroupsMap['Names'] = []
 353     FunctionalGroupsMap['SMARTSPattern'] = {}
 354     
 355     RetrieveDataFromFunctionalGroupsHierarchy(FunctionalGroupsNodes)
 356 
 357     if not len(FunctionalGroupsMap['Names']):
 358         MiscUtil.PrintError("Failed to retrieve any functional group names and SMARTS patterns...")
 359         
 360     MiscUtil.PrintInfo("Total number of functional groups present functional group hierarchy: %d" % (len(FunctionalGroupsMap['Names'])))
 361     
 362 def RetrieveDataFromFunctionalGroupsHierarchy(FGNodes):
 363     """Retrieve functional groups data by recursively visiting functional group nodes."""
 364     
 365     for FGNode in FGNodes:
 366         Name = FGNode.label
 367         SMARTSPattern = FGNode.smarts
 368 
 369         if Name in FunctionalGroupsMap['SMARTSPattern']:
 370             MiscUtil.PrintWarning("Ignoring duplicate functional group name: %s..." % Name)
 371         else:
 372             FunctionalGroupsMap['Names'].append(Name)
 373             FunctionalGroupsMap['SMARTSPattern'][Name] = SMARTSPattern
 374 
 375         RetrieveDataFromFunctionalGroupsHierarchy(FGNode.children)
 376 
 377 def ListFunctionalGroupsInfo():
 378     """List functional groups information"""
 379 
 380     MiscUtil.PrintInfo("\nListing available functional groups names and SMARTS patterns...")
 381     MiscUtil.PrintInfo("\nFunctionalGroupName\tSMARTSPattern")
 382     
 383     for Name in sorted(FunctionalGroupsMap['Names']):
 384         SMARTSPattern = FunctionalGroupsMap['SMARTSPattern'][Name]
 385         MiscUtil.PrintInfo("%s\t%s" % (Name, SMARTSPattern))
 386     
 387     MiscUtil.PrintInfo("")

 399     OptionsInfo["CombineMatchResults"] = True

 401         OptionsInfo["CombineMatchResults"] = False
 402         if Options["--outfile"]:

 404             OptionsInfo["OutfileBasename"] = FileName
 405             OptionsInfo["OutfileExt"] = FileExt
 406     
 407     OptionsInfo["CombineOperator"] = Options["--combineOperator"]
 408     OptionsInfo["AndCombineOperatorMode"] = True
 409     if re.match("^or$", Options["--combineOperator"], re.I):
 410         OptionsInfo["AndCombineOperatorMode"] = False
 411     
 412     OptionsInfo["GroupNamesFile"] = None
 413     if not re.match("^auto$", Options["--groupNamesFile"], re.I):
 414         OptionsInfo["GroupNamesFile"] = Options["--groupNamesFile"]
 415         
 416     OptionsInfo["FunctionalGroups"] = Options["--functionalGroups"]
 417     
 418     OptionsInfo["Infile"] = Options["--infile"]
 419     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], Options["--infile"])
 420     
 421     OptionsInfo["Outfile"] = Options["--outfile"]
 422     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"], Options["--infile"], Options["--outfile"])
 423     
 424     OptionsInfo["Overwrite"] = Options["--overwrite"]
 425     
 426     OptionsInfo["CountMode"] = False
 427     if re.match("^count$", Options["--mode"], re.I):
 428         OptionsInfo["CountMode"] = True
 429     
 430     OptionsInfo["UseChirality"] = False
 431     if re.match("^yes$", Options["--useChirality"], re.I):
 432         OptionsInfo["UseChirality"] = True

 434 def RetrieveOptions():
 435     """Retrieve command line arguments and options"""
 436     
 437     # Get options...
 438     global Options
 439     Options = docopt(_docoptUsage_)
 440 
 441     # Set current working directory to the specified directory...
 442     WorkingDir = Options["--workingdir"]
 443     if WorkingDir:
 444         os.chdir(WorkingDir)
 445     
 446     # Handle examples option...
 447     if "--examples" in Options and Options["--examples"]:
 448         MiscUtil.PrintInfo(MiscUtil.GetExamplesTextFromDocOptText(_docoptUsage_))

 450     
 451     # Handle listing of functional group information...
 452     if  Options and Options["--list"]:
 453         ProcessListFunctionalGroupsOption()
 454         sys.exit(0)

 457     """Process list functional groups information."""

 468 def ValidateOptions():
 469     """Validate option values"""
 470     
 471     MiscUtil.ValidateOptionTextValue("-c, --combineMatches", Options["--combineMatches"], "yes no")
 472     MiscUtil.ValidateOptionTextValue("--combineOperator", Options["--combineOperator"], "and or")
 473     
 474     if not re.match("^auto$", Options["--groupNamesFile"], re.I):
 475         MiscUtil.ValidateOptionFilePath("-g, groupNamesFile", Options["--groupNamesFile"])
 476         
 477     if re.match("^none$", Options["--functionalGroups"], re.I):
 478         MiscUtil.PrintError("The name(s) of functional groups must be specified using \"-f, --functionalGroups\" option")
 479         
 480     MiscUtil.ValidateOptionFilePath("-i, --infile", Options["--infile"])
 481     MiscUtil.ValidateOptionFileExt("-i, --infile", Options["--infile"], "sdf sd smi txt csv tsv")
 482     if Options["--outfile"]:
 483         MiscUtil.ValidateOptionFileExt("-o, --outfile", Options["--outfile"], "sdf sd smi")
 484         MiscUtil.ValidateOptionsOutputFileOverwrite("-o, --outfile", Options["--outfile"], "--overwrite", Options["--overwrite"])
 485         MiscUtil.ValidateOptionsDistinctFileNames("-i, --infile", Options["--infile"], "-o, --outfile", Options["--outfile"])
 486     
 487     MiscUtil.ValidateOptionTextValue("-m, --mode", Options["--mode"], "retrieve count")
 488     if re.match("^retrieve$", Options["--mode"], re.I):
 489         if not Options["--outfile"]:
 490             MiscUtil.PrintError("The outfile must be specified using \"-o, --outfile\" during \"retrieve\" value of \"-m, --mode\" option")
 491         
 492     MiscUtil.ValidateOptionTextValue("--useChirality", Options["--useChirality"], "yes no")
